local function GetCorrectAnimID(char, packString, defR15, defR6)
    local aniStruct = char:FindFirstChild("Animate") or char:FindFirstChild("animate")
    if aniStruct then local folder = aniStruct:FindFirstChild(packString) if folder then local animItem = folder:FindFirstChildWhichIsA("Animation") if animItem and animItem.AnimationId ~= "" then return animItem.AnimationId end end end
    local hum = char:FindFirstChild("Humanoid")
    if hum and hum.RigType == Enum.HumanoidRigType.R6 then return defR6 end
    return defR15
end

local function PlaySyncAnims(trackName) for _, arr in pairs({FakeAnims, RealAnims}) do if arr[trackName] and not arr[trackName].IsPlaying then arr[trackName]:Play(0.1) end end end
local function StopSyncAnims() for _, arr in pairs({FakeAnims, RealAnims}) do for _, t in pairs(arr) do if t and t.IsPlaying then t:Stop(0.1) end end end end

local function ShutdownGhostSystemGracefully()
    for _, conn in pairs(FakelagHooks) do conn:Disconnect() end; table.clear(FakelagHooks); StopSyncAnims()
    if FakeChar then
        local targetReturnCF = FakeChar:GetPrimaryPartCFrame(); FakeChar:Destroy(); FakeChar = nil
        if RealChar then
            local realHumTarget = RealChar:FindFirstChild("Humanoid")
            if realHumTarget then realHumTarget.WalkSpeed = 16; realHumTarget.JumpPower = 50; realHumTarget.PlatformStand = false end
            RealChar:PivotTo(targetReturnCF) 
            local realRootTarget = RealChar:FindFirstChild("HumanoidRootPart")
            if realRootTarget then realRootTarget.Velocity = Vector3.zero; realRootTarget.Anchored = false end
            for item, originVal in pairs(RealPartOriginalStates) do if item and item.Parent then item.Transparency = originVal end end
            local animateScript = RealChar:FindFirstChild("Animate") or RealChar:FindFirstChild("animate")
            if animateScript and animateScript:IsA("LocalScript") then animateScript.Enabled = true end
            table.clear(RealPartOriginalStates); table.clear(RealAnims); table.clear(FakeAnims)
            Camera.CameraSubject = RealChar:FindFirstChild("Humanoid")
        end
    end
end

local function ActivateFakelagGhost()
    RealChar = LocalPlayer.Character
    if not RealChar or not RealChar.PrimaryPart or not RealChar:FindFirstChild("HumanoidRootPart") then return false end
    
    CurrentGhostPosServerCacheCF = RealChar.PrimaryPart.CFrame; RealChar.Archivable = true; FakeChar = RealChar:Clone(); FakeChar:PivotTo(CurrentGhostPosServerCacheCF); FakeChar.Name = "ReflexLocalClientGhost"
    
    local realHum = RealChar:FindFirstChild("Humanoid"); local realRoot = RealChar:FindFirstChild("HumanoidRootPart")
    if realHum then realHum.WalkSpeed = 0; realHum.JumpPower = 0 end; if realRoot then realRoot.Anchored = false end
    
    local origAnimateScript = RealChar:FindFirstChild("Animate") or RealChar:FindFirstChild("animate"); if origAnimateScript and origAnimateScript:IsA("LocalScript") then origAnimateScript.Enabled = false end

    local solidPartsCache = {}; table.clear(RealPartOriginalStates)
    for _, origDescendant in ipairs(RealChar:GetDescendants()) do
        if origDescendant:IsA("BasePart") or origDescendant:IsA("Decal") or origDescendant:IsA("Texture") then
            RealPartOriginalStates[origDescendant] = origDescendant.Transparency
            if origDescendant.Name == "HumanoidRootPart" then origDescendant.Transparency = 1 else origDescendant.Transparency = 0.5 end
            if origDescendant:IsA("BasePart") then table.insert(solidPartsCache, origDescendant) end
        end
    end

    for _, fakeDescendant in ipairs(FakeChar:GetDescendants()) do
        if fakeDescendant:IsA("BaseScript") then fakeDescendant:Destroy() end
        if fakeDescendant:IsA("BasePart") then
            fakeDescendant.Anchored = false; fakeDescendant.CanCollide = (fakeDescendant.Name == "HumanoidRootPart"); if fakeDescendant.Name == "HumanoidRootPart" then fakeDescendant.Transparency = 1 end
            for _, mappedPart in ipairs(solidPartsCache) do local noCollision = Instance.new("NoCollisionConstraint"); noCollision.Part0 = mappedPart; noCollision.Part1 = fakeDescendant; noCollision.Parent = fakeDescendant end
        end
    end

    FakeChar.Parent = workspace
    local fakeHum = FakeChar:FindFirstChild("Humanoid")
    local idleSrc = Instance.new("Animation"); idleSrc.AnimationId = GetCorrectAnimID(RealChar, "idle", "rbxassetid://507766388", "rbxassetid://180435571")
    local walkSrc = Instance.new("Animation"); walkSrc.AnimationId = GetCorrectAnimID(RealChar, "walk", "rbxassetid://507777826", "rbxassetid://180426354")
    local runSrc  = Instance.new("Animation"); runSrc.AnimationId  = GetCorrectAnimID(RealChar, "run",  "rbxassetid://507767714", "rbxassetid://180426354"); if runSrc.AnimationId == "" then runSrc.AnimationId = walkSrc.AnimationId end
    local jpSrc   = Instance.new("Animation"); jpSrc.AnimationId   = GetCorrectAnimID(RealChar, "jump", "rbxassetid://507765000", "rbxassetid://125750702")
    local fallSrc = Instance.new("Animation"); fallSrc.AnimationId = GetCorrectAnimID(RealChar, "freefall", "rbxassetid://507767968", "rbxassetid://180436148")

    local fakeAnimator = fakeHum:FindFirstChildOfClass("Animator") or Instance.new("Animator", fakeHum); FakeAnims.Idle = fakeAnimator:LoadAnimation(idleSrc); FakeAnims.Run = fakeAnimator:LoadAnimation(runSrc); FakeAnims.Fall = fakeAnimator:LoadAnimation(fallSrc); FakeAnims.Jump = fakeAnimator:LoadAnimation(jpSrc)
    local rAnimator = realHum:FindFirstChildOfClass("Animator") or Instance.new("Animator", realHum); RealAnims.Idle = rAnimator:LoadAnimation(idleSrc); RealAnims.Run = rAnimator:LoadAnimation(runSrc); RealAnims.Fall = rAnimator:LoadAnimation(fallSrc); RealAnims.Jump = rAnimator:LoadAnimation(jpSrc)
    
    Camera.CameraSubject = fakeHum; PlaySyncAnims("Idle")
    table.insert(FakelagHooks, fakeHum.StateChanged:Connect(function(_, newState)
        if newState == Enum.HumanoidStateType.Jumping then StopSyncAnims(); PlaySyncAnims("Jump") elseif newState == Enum.HumanoidStateType.Freefall then StopSyncAnims(); PlaySyncAnims("Fall") elseif newState == Enum.HumanoidStateType.Landed then StopSyncAnims(); PlaySyncAnims("Idle") end
    end))
    table.insert(FakelagHooks, RunService.Stepped:Connect(function()
        if fakelagSystemActiveNow and RealChar and RealChar.PrimaryPart then RealChar.PrimaryPart.CFrame = CurrentGhostPosServerCacheCF; RealChar.PrimaryPart.AssemblyLinearVelocity = Vector3.zero; RealChar.PrimaryPart.AssemblyAngularVelocity = Vector3.zero end
    end))
    table.insert(FakelagHooks, UserInputService.JumpRequest:Connect(function() if fakelagSystemActiveNow and fakeHum and fakeHum:GetState() ~= Enum.HumanoidStateType.Freefall then fakeHum.Jump = true end end))

    fakelagAccumulator = 0; return true
end

local function processFakelagEngineRenderCycle(dt)
    if Config.fakelag and fakelagKeyPressed and not fakelagSystemActiveNow then
        if ActivateFakelagGhost() then fakelagSystemActiveNow = true end
    elseif (not Config.fakelag or not fakelagKeyPressed) and fakelagSystemActiveNow then
        ShutdownGhostSystemGracefully(); fakelagSystemActiveNow = false
    end
    if fakelagSystemActiveNow and FakeChar then
        fakelagAccumulator = fakelagAccumulator + dt
        if fakelagAccumulator >= Config.fakelag_delay then fakelagAccumulator = 0; if FakeChar.PrimaryPart then CurrentGhostPosServerCacheCF = FakeChar.PrimaryPart.CFrame end end
        
        local fakeHum = FakeChar:FindFirstChild("Humanoid")
        if fakeHum then 
            fakeHum.WalkSpeed = Config.fakelag_speed; local inputVector = Vector3.zero
            if Controls then inputVector = Controls:GetMoveVector() else
                if UserInputService:IsKeyDown(Enum.KeyCode.W) then inputVector += Vector3.new(0, 0, -1) end
                if UserInputService:IsKeyDown(Enum.KeyCode.S) then inputVector += Vector3.new(0, 0, 1) end
                if UserInputService:IsKeyDown(Enum.KeyCode.A) then inputVector += Vector3.new(-1, 0, 0) end
                if UserInputService:IsKeyDown(Enum.KeyCode.D) then inputVector += Vector3.new(1, 0, 0) end
            end
            
            local targetVector = (Camera.CFrame.RightVector * inputVector.X) + (Camera.CFrame.LookVector * -inputVector.Z)
            targetVector = Vector3.new(targetVector.X, 0, targetVector.Z); local fakeState = fakeHum:GetState()
            local GrndChkM = (fakeState == Enum.HumanoidStateType.Running or fakeState == Enum.HumanoidStateType.RunningNoPhysics)
            
            if targetVector.Magnitude > 0 then
                fakeHum:Move(targetVector.Unit, false); if GrndChkM and FakeAnims.Run and not FakeAnims.Run.IsPlaying then StopSyncAnims(); PlaySyncAnims("Run") end
            else
                fakeHum:Move(Vector3.zero, false); if GrndChkM and FakeAnims.Idle and not FakeAnims.Idle.IsPlaying then StopSyncAnims(); PlaySyncAnims("Idle") end
            end
        end
    end
end

local function capturePlayerPart(part)
    if PlayerSkinOriginals.parts[part] then return end; PlayerSkinOriginals.parts[part] = { color = part.Color, material = part.Material }
    local decalList = {}
    for _, child in ipairs(part:GetChildren()) do
        if child:IsA("Decal") or child:IsA("Texture") or child:IsA("SurfaceAppearance") then
            local info = { obj = child, transparency = child.Transparency }
            if child:IsA("Decal") or child:IsA("Texture") then info.texture = child.Texture elseif child:IsA("SurfaceAppearance") then info.colorMap = child.ColorMap; info.normalMap = child.NormalMap; info.metalnessMap = child.MetalnessMap; info.roughnessMap = child.RoughnessMap end
            table.insert(decalList, info); child.Transparency = 1
            pcall(function() if child:IsA("Decal") or child:IsA("Texture") then child.Texture = "" elseif child:IsA("SurfaceAppearance") then child.ColorMap = ""; child.NormalMap = ""; child.MetalnessMap = ""; child.RoughnessMap = "" end end)
        end
    end
    if next(decalList) then PlayerSkinOriginals.decals[part] = decalList end
end

local function passesTeamCheck(targetPlayer)
    if not Config.team_check then return true end
    local tm = targetPlayer.Team; if not tm then return true end; local nm = string.lower(tm.Name)
    local isGuard = string.find(nm, "guard") or string.find(nm, "cop") or string.find(nm, "police")
    local isInmate = string.find(nm, "inmate") or string.find(nm, "prisoner")
    local isCrim = string.find(nm, "criminal")
    
    if Config.ignore_guards and isGuard then return false end; if Config.ignore_criminals and isCrim then return false end
    if Config.ignore_inmates and isInmate then 
        local targetChar = targetPlayer.Character
        if targetChar then
            if Config.target_hostile and targetChar:GetAttribute("Hostile") then return true end
            if Config.target_trespassing and targetChar:GetAttribute("Trespassing") then if isLocalHoldingTaser() then return true end end
        end
        return false 
    end
    return true
end

    if not Config.wall_check then return true end
    local root = character:FindFirstChild("HumanoidRootPart"); if not root then return false end

local function getClosestTarget()
    local closestPlayer = nil; local shortestDistance = math.huge; local mousePos = UserInputService:GetMouseLocation()
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            if not passesTeamCheck(player) then continue end
            local character = player.Character; local humanoid = character and character:FindFirstChildOfClass("Humanoid")
            if character and humanoid and humanoid.Health > 0 then
                local targetPart = character:FindFirstChild(Config.aimbot_target)
                if targetPart then
                    local screenPos, onScreen = Camera:WorldToViewportPoint(targetPart.Position)
                    if onScreen then
                        local distance = (Vector2.new(screenPos.X, screenPos.Y) - mousePos).Magnitude
                        if distance < Config.aimbot_radius and distance < shortestDistance then
                            if checkVisibility(player, character) then shortestDistance = distance; closestPlayer = player end
                        end
                    end
                end
            end
        end
    end
    return closestPlayer
end

local function aimAtTarget(targetPlayer)
    if not targetPlayer or not targetPlayer.Character then return end
    local targetPart = targetPlayer.Character:FindFirstChild(Config.aimbot_target)
    if targetPart then Camera.CFrame = CFrame.new(Camera.CFrame.Position, targetPart.Position) end
end

local function noUpvals(fn) return function(...) return fn(...) end end
local origCastRay; local hooked = false
local function setupHook()
    local castRayFunc = nil
    if type(filtergc) == "function" then castRayFunc = filtergc("function", {Name = "castRay"}, true)
    elseif type(getgc) == "function" then for _, v in pairs(getgc(true)) do if type(v) == "function" and debug.getinfo(v).name == "castRay" then castRayFunc = v break end end end
    if not castRayFunc then return false end

    origCastRay = hookfunction(castRayFunc, noUpvals(function(startPos, targetPos, ...)
        if not checkcaller() and aimbotKeyPressed and Config.silent_aim and silentAimTarget and silentAimTarget.Character then
            local tgtPart = silentAimTarget.Character:FindFirstChild(Config.aimbot_target) or silentAimTarget.Character:FindFirstChild("Head")
            if tgtPart then return tgtPart, tgtPart.Position end
        end
        return origCastRay(startPos, targetPos, ...)
    end))
    return true
end
task.spawn(function() while not hooked do if setupHook() then hooked = true end task.wait(0.5) end end)

end

local function getDynamicBoundingBox(character)
    local minX, minY, maxX, maxY = math.huge, math.huge, -math.huge, -math.huge; local anyOnScreen = false
    for _, part in ipairs(character:GetChildren()) do
        if part:IsA("BasePart") then
            local size = part.Size; local cf = part.CFrame
            local vertices = { cf * Vector3.new(-size.X/2, -size.Y/2, -size.Z/2), cf * Vector3.new(size.X/2, -size.Y/2, -size.Z/2), cf * Vector3.new(-size.X/2, size.Y/2, -size.Z/2), cf * Vector3.new(-size.X/2, -size.Y/2, size.Z/2), cf * Vector3.new(size.X/2, size.Y/2, -size.Z/2), cf * Vector3.new(size.X/2, -size.Y/2, size.Z/2), cf * Vector3.new(-size.X/2, size.Y/2, size.Z/2), cf * Vector3.new(size.X/2, size.Y/2, size.Z/2) }
            for i = 1, 8 do local pos, onScreen = Camera:WorldToViewportPoint(vertices[i]) if onScreen then anyOnScreen = true end if pos.X < minX then minX = pos.X end if pos.X > maxX then maxX = pos.X end if pos.Y < minY then minY = pos.Y end if pos.Y > maxY then maxY = pos.Y end end
        end
    end
    if not anyOnScreen then return nil end; local w = maxX - minX; local h = maxY - minY
    if w < 5 or h < 5 then return nil end; return minX, minY, w, h
end

        if player ~= LocalPlayer then
            local data = espStorage[player]; local character = player.Character; local humanoid = character and character:FindFirstChildOfClass("Humanoid")
            if character and humanoid and data then
                updatePlayerChams(player, character); local posX, posY, sizeX, sizeY = getDynamicBoundingBox(character)
                if posX then

                    else data.healthLabel.Visible = false end
                else data.container.Visible = false; data.nameLabel.Visible = false; data.healthLabel.Visible = false end

local activeColorKey, activeAlphaKey, activePreviewRect = nil, nil, nil
local currentH, currentS, currentV, currentAlpha = 0, 0, 0, 0

local function updatePaletteColor(skipHexBoxUpdate)
    if not activeColorKey then return end
    local selectedColor = Color3.fromHSV(currentH, currentS, currentV); Config[activeColorKey] = selectedColor; Config[activeAlphaKey] = currentAlpha
    alphaBgGradient.Color = ColorSequence.new({ColorSequenceKeypoint.new(0, selectedColor), ColorSequenceKeypoint.new(1, selectedColor)})
    if activePreviewRect then activePreviewRect.BackgroundColor3 = selectedColor; activePreviewRect.BackgroundTransparency = currentAlpha end
    if not skipHexBoxUpdate then hexBox.Text = colorToHex(selectedColor) end
    if activeColorKey == "menu_color" then applyTheme() elseif activeColorKey == "lighting_color" then updateLighting() elseif activeColorKey == "fog_color" then updateAtmosphere() elseif string.sub(activeColorKey, 1, 11) == "skin_color_" then updateWeaponSkins() elseif activeColorKey == "player_skin_color" then updatePlayerSkin() end
end

local function bindExternalColorPicker(colKey, alphaKey, previewRect, contextText)
    if activeColorKey == colKey then ColorPickerPanel.Visible = not ColorPickerPanel.Visible; if not ColorPickerPanel.Visible then activeColorKey = nil; activeAlphaKey = nil; activePreviewRect = nil end return end
    activeColorKey = colKey; activeAlphaKey = alphaKey; activePreviewRect = previewRect; CPIndicator.Text = contextText and contextText:upper() or "SELECTING COLOR"; ColorPickerPanel.Visible = true
    local col = Config[colKey] or Color3.fromRGB(255, 255, 255); currentAlpha = Config[alphaKey] or 0; currentH, currentS, currentV = Color3.toHSV(col)
    colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1); baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1)); hueSlider.Position = UDim2.new(0, -2, currentH, 0); pin.Position = UDim2.new(currentS, 0, 1 - currentV, 0)
    alphaSlider.Position = UDim2.new(1 - currentAlpha, -1, 0, -2); hexBox.Text = colorToHex(col); alphaBgGradient.Color = ColorSequence.new({ColorSequenceKeypoint.new(0, col), ColorSequenceKeypoint.new(1, col)})
end

local dragCanvas, dragHue, dragAlpha = false, false, false
canvas.InputBegan:Connect(function(input) if not activeColorKey then return end if input.UserInputType == Enum.UserInputType.MouseButton1 then dragCanvas = true; local relativeX = math.clamp((input.Position.X - canvas.AbsolutePosition.X) / canvas.AbsoluteSize.X, 0, 1); local relativeY = math.clamp((input.Position.Y - canvas.AbsolutePosition.Y) / canvas.AbsoluteSize.Y, 0, 1); currentS = relativeX; currentV = 1 - relativeY; pin.Position = UDim2.new(relativeX, 0, relativeY, 0); updatePaletteColor(false) end end)
hueStrip.InputBegan:Connect(function(input) if not activeColorKey then return end if input.UserInputType == Enum.UserInputType.MouseButton1 then dragHue = true; local relativeY = math.clamp((input.Position.Y - hueStrip.AbsolutePosition.Y) / hueStrip.AbsoluteSize.Y, 0, 1); currentH = relativeY; hueSlider.Position = UDim2.new(0, -2, relativeY, 0); colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1); baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1)); updatePaletteColor(false) end end)
alphaStrip.InputBegan:Connect(function(input) if not activeColorKey then return end if input.UserInputType == Enum.UserInputType.MouseButton1 then dragAlpha = true; local relativeX = math.clamp((input.Position.X - alphaStrip.AbsolutePosition.X) / alphaStrip.AbsoluteSize.X, 0, 1); currentAlpha = 1 - relativeX; alphaSlider.Position = UDim2.new(relativeX, -1, 0, -2); updatePaletteColor(false) end end)
hexBox.FocusLost:Connect(function() if not activeColorKey then return end local targetColor = hexToColor(hexBox.Text) if targetColor then currentH, currentS, currentV = Color3.toHSV(targetColor); colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1); baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1)); hueSlider.Position = UDim2.new(0, -2, currentH, 0); pin.Position = UDim2.new(currentS, 0, 1 - currentV, 0); updatePaletteColor(true); hexBox.Text = colorToHex(targetColor) else hexBox.Text = colorToHex(Config[activeColorKey]) end end)
UserInputService.InputEnded:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then dragCanvas = false; dragHue = false; dragAlpha = false end end)
UserInputService.InputChanged:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseMovement then
        if dragCanvas and ColorPickerPanel.Visible then local relativeX = math.clamp((input.Position.X - canvas.AbsolutePosition.X) / canvas.AbsoluteSize.X, 0, 1); local relativeY = math.clamp((input.Position.Y - canvas.AbsolutePosition.Y) / canvas.AbsoluteSize.Y, 0, 1); currentS = relativeX; currentV = 1 - relativeY; pin.Position = UDim2.new(relativeX, 0, relativeY, 0); updatePaletteColor(false)
        elseif dragHue and ColorPickerPanel.Visible then local relativeY = math.clamp((input.Position.Y - hueStrip.AbsolutePosition.Y) / hueStrip.AbsoluteSize.Y, 0, 1); currentH = relativeY; hueSlider.Position = UDim2.new(0, -2, relativeY, 0); colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1); baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1)); updatePaletteColor(false)
        elseif dragAlpha and ColorPickerPanel.Visible then local relativeX = math.clamp((input.Position.X - alphaStrip.AbsolutePosition.X) / alphaStrip.AbsoluteSize.X, 0, 1); currentAlpha = 1 - relativeX; alphaSlider.Position = UDim2.new(relativeX, -1, 0, -2); updatePaletteColor(false) end
    end
end)

local transitionActive = false
local function toggleMenu()
    if transitionActive then return end; transitionActive = true; menuVisible = not menuVisible
    if menuVisible then
        MainFrame.Visible = true; TweenService:Create(MainFrame, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()

    else
        TweenService:Create(MainFrame, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play(); TweenService:Create(Sidebar, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play(); TweenService:Create(Logo, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {TextTransparency = 1}):Play(); TweenService:Create(MainStroke, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {Transparency = 1}):Play(); TweenService:Create(SepLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play(); TweenService:Create(AccentLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()

    if listeningForAimbotBind then
        local bound = nil
        if input.UserInputType == Enum.UserInputType.Keyboard and input.KeyCode ~= Enum.KeyCode.Unknown then bound = input.KeyCode elseif input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.MouseButton2 or input.UserInputType == Enum.UserInputType.MouseButton3 then bound = input.UserInputType end
        if bound then Config.aimbot_key = bound; UIRefs.AimbotBindBtn.Text = getBindName(bound); UIRefs.AimbotBindBtn.TextColor3 = Config.menu_color; listeningForAimbotBind = false; aimbotKeyPressed = false; return end
    end

    if listeningForFakelagBind then
        local bound = nil
        if input.UserInputType == Enum.UserInputType.Keyboard and input.KeyCode ~= Enum.KeyCode.Unknown then bound = input.KeyCode elseif input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.MouseButton2 or input.UserInputType == Enum.UserInputType.MouseButton3 then bound = input.UserInputType end

    end

    if processed then return end
    if input.KeyCode == Config.toggleKey then toggleMenu() end
    if matchesBind(input, Config.aimbot_key) then if Config.aimbot_mode == "Hold" then aimbotKeyPressed = true elseif Config.aimbot_mode == "Toggle" then aimbotKeyPressed = not aimbotKeyPressed end end
    if matchesBind(input, Config.fakelag_key) then if Config.fakelag_mode == "Hold" then fakelagKeyPressed = true elseif Config.fakelag_mode == "Toggle" then fakelagKeyPressed = not fakelagKeyPressed end end
