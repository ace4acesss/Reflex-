local function checkVisibility(player, character)
    local root = character:FindFirstChild("HumanoidRootPart") if not root then return false end
    local params = RaycastParams.new() params.FilterType = Enum.RaycastFilterType.Exclude local filterTargets = {character}
    if LocalPlayer.Character then table.insert(filterTargets, LocalPlayer.Character) end params.FilterDescendantsInstances = filterTargets
    local result = workspace:Raycast(Camera.CFrame.Position, root.Position - Camera.CFrame.Position, params)
    return not result
end

local function checkTeamWhitelisted(player)
    if not Config.team_check then return true end
    if player.Team then
        local t = string.lower(player.Team.Name)
        if (t == "guards" or t == "guard" or t == "police") and Config.ignore_guards then return false end
        if (t == "inmates" or t == "inmate") and Config.ignore_inmates then return false end
        if (t == "criminals" or t == "criminal") and Config.ignore_criminals then return false end
    end
    return true
end

local function getClosestTarget()
    local closestPlayer = nil local shortestDistance = math.huge local mousePos = UserInputService:GetMouseLocation()
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and checkTeamWhitelisted(player) then
            local character = player.Character local humanoid = character and character:FindFirstChildOfClass("Humanoid")
            if character and humanoid and humanoid.Health > 0 then
                local targetPart = character:FindFirstChild(Config.aimbot_target)
                if targetPart then
                    local screenPos, onScreen = Camera:WorldToViewportPoint(targetPart.Position)
                    if onScreen then
                        local distance = (Vector2.new(screenPos.X, screenPos.Y) - mousePos).Magnitude
                        if distance < Config.aimbot_radius and distance < shortestDistance and (not Config.aimbot_wall_check or checkVisibility(player, character)) then shortestDistance, closestPlayer = distance, player end
                    end
                end
            end
        end
    end
    return closestPlayer
end

local function aimAtTarget(targetPlayer)
    if not targetPlayer or not targetPlayer.Character then return end
    local targetPart = targetPlayer.Character:FindFirstChild(Config.aimbot_target)
    if targetPart then Camera.CFrame = CFrame.new(Camera.CFrame.Position, targetPart.Position) end
end

local function processAimbotTargetLoop()
    if (Config.aimbot or Config.silent_aim) and Config.draw_radius then
        local mousePos = UserInputService:GetMouseLocation() FOVCircle.Position = UDim2.new(0, mousePos.X, 0, mousePos.Y) FOVCircle.Size = UDim2.new(0, Config.aimbot_radius * 2, 0, Config.aimbot_radius * 2) FOVCircle.Visible = true FOVStroke.Color, FOVStroke.Transparency = Config.draw_radius_color, Config.draw_radius_transparency
    else FOVCircle.Visible = false end
    if Config.aimbot and aimbotKeyPressed then local target = getClosestTarget() if target then aimAtTarget(target) end end
end

local oldFireServerHooksInstalled, oldFireServer = false, nil
local function injectSilentAimHookFunction()
    if oldFireServerHooksInstalled then return end
    pcall(function()
        local rInstanceEvent = Instance.new("RemoteEvent")
        oldFireServer = hookfunction(rInstanceEvent.FireServer, function(self, ...)
            local args = {...}
            if Config.silent_aim and aimbotKeyPressed then
                if typeof(self) == "Instance" and self:IsA("RemoteEvent") and self.Name == "ShootEvent" then
                    local targetPlayerInfoObj = getClosestTarget()
                    if targetPlayerInfoObj and targetPlayerInfoObj.Character then
                        local pCharacterMeshTargettings = targetPlayerInfoObj.Character:FindFirstChild(Config.aimbot_target)
                        if pCharacterMeshTargettings and type(args[1]) == "table" then
                            for _, fired_projectilies in ipairs(args[1]) do
                                if type(fired_projectilies) == "table" then fired_projectilies.Hit = pCharacterMeshTargettings fired_projectilies.Cframe = CFrame.new(Camera.CFrame.Position, pCharacterMeshTargettings.Position) fired_projectilies.Distance = (Camera.CFrame.Position - pCharacterMeshTargettings.Position).Magnitude end
                            end
                        end
                    end
                end
            end
            return oldFireServer(self, unpack(args))
        end)
        oldFireServerHooksInstalled = true
    end)
end

local function getMaterialByName(materialName) local name = tostring(materialName) local ok, material = pcall(function() return Enum.Material[name] end) if ok and material then return material end for _, m in ipairs(Enum.Material:GetEnumItems()) do if string.lower(m.Name) == string.lower(name) then return m end end return Enum.Material.Plastic end
local function getWeaponRoots() local roots = {} local backpack = LocalPlayer:FindFirstChildOfClass("Backpack") if backpack then table.insert(roots, backpack) end if LocalPlayer.Character then table.insert(roots, LocalPlayer.Character) end return roots end
local function findWeaponInstances(weaponName) local found = {} for _, root in ipairs(getWeaponRoots()) do for _, item in ipairs(root:GetDescendants()) do if string.lower(item.Name) == string.lower(weaponName) and (item:IsA("Tool") or item:IsA("Model")) then table.insert(found, item) end end end return found end
local function captureOriginal(weaponName, meshObject, paintPart) local originals = WeaponSkinOriginals[weaponName] if originals[meshObject] then return end local textureProperty = meshObject:IsA("MeshPart") and "TextureID" or "TextureId" local textureValue = "" pcall(function() textureValue = meshObject[textureProperty] end) originals[meshObject] = { textureProperty = textureProperty, texture = textureValue, part = paintPart, color = paintPart and paintPart.Color or nil, material = paintPart and paintPart.Material or nil } end

local function applySkinToMesh(weaponName, meshObject)
    local id, paintPart = WeaponSkinIds[weaponName], meshObject:IsA("BasePart") and meshObject or meshObject.Parent if not paintPart or not paintPart:IsA("BasePart") then return end
    captureOriginal(weaponName, meshObject, paintPart) pcall(function() if meshObject:IsA("MeshPart") then meshObject.TextureID = "" else meshObject.TextureId = "" end end) paintPart.Color, paintPart.Material = Config["skin_color_" .. id], getMaterialByName(Config["skin_material_" .. id])
end

local function restoreWeaponSkin(weaponName)
    local originals = WeaponSkinOriginals[weaponName] if not originals then return end
    for meshObject, data in pairs(originals) do if meshObject and meshObject.Parent then pcall(function() meshObject[data.textureProperty] = data.texture or "" end) if data.part and data.part.Parent then if data.color then data.part.Color = data.color end if data.material then data.part.Material = data.material end end end end
    table.clear(originals)
end

end

local PlayerSkinOriginals = { clothes = {}, parts = {}, meshes = {}, decals = {} }
local function clearPlayerSkinCache() table.clear(PlayerSkinOriginals.clothes) table.clear(PlayerSkinOriginals.parts) table.clear(PlayerSkinOriginals.meshes) table.clear(PlayerSkinOriginals.decals) end
local function restorePlayerSkin()
    for clothing, parent in pairs(PlayerSkinOriginals.clothes) do if clothing and not clothing.Parent and parent and parent.Parent then clothing.Parent = parent end end
    for part, data in pairs(PlayerSkinOriginals.parts) do
        if part and part.Parent then part.Color, part.Material = data.color, data.material end
        local decalList = PlayerSkinOriginals.decals[part] if decalList then for _, info in ipairs(decalList) do if info.obj and info.obj.Parent then info.obj.Transparency = info.transparency pcall(function() if info.obj:IsA("Decal") or info.obj:IsA("Texture") then info.obj.Texture = info.texture elseif info.obj:IsA("SurfaceAppearance") then info.obj.ColorMap, info.obj.NormalMap, info.obj.MetalnessMap, info.obj.RoughnessMap = info.colorMap, info.normalMap, info.metalnessMap, info.roughnessMap end end) end end end
    end
    for meshObject, data in pairs(PlayerSkinOriginals.meshes) do if meshObject and meshObject.Parent then pcall(function() meshObject[data.textureProperty] = data.texture or "" end) end end clearPlayerSkinCache()
end

local function capturePlayerPart(part)
    if PlayerSkinOriginals.parts[part] then return end PlayerSkinOriginals.parts[part] = { color = part.Color, material = part.Material } local decalList = {}
    for _, child in ipairs(part:GetChildren()) do
        if child:IsA("Decal") or child:IsA("Texture") or child:IsA("SurfaceAppearance") then
            local info = { obj = child, transparency = child.Transparency }
            if child:IsA("Decal") or child:IsA("Texture") then info.texture = child.Texture elseif child:IsA("SurfaceAppearance") then info.colorMap, info.normalMap, info.metalnessMap, info.roughnessMap = child.ColorMap, child.NormalMap, child.MetalnessMap, child.RoughnessMap end
            table.insert(decalList, info) child.Transparency = 1 pcall(function() if child:IsA("Decal") or child:IsA("Texture") then child.Texture = "" elseif child:IsA("SurfaceAppearance") then child.ColorMap, child.NormalMap, child.MetalnessMap, child.RoughnessMap = "", "", "", "" end end)
        end
    end
    if next(decalList) then PlayerSkinOriginals.decals[part] = decalList end
end

local function capturePlayerMesh(meshObject)
    if PlayerSkinOriginals.meshes[meshObject] then return end
    local textureProperty = meshObject:IsA("MeshPart") and "TextureID" or (meshObject:IsA("SpecialMesh") and "TextureId" or nil)
    if not textureProperty then return end local textureValue = "" pcall(function() textureValue = meshObject[textureProperty] end)
    PlayerSkinOriginals.meshes[meshObject] = { textureProperty = textureProperty, texture = textureValue }
end

local function applyPlayerSkin()
    local character = LocalPlayer.Character if not character then return end
    for _, item in ipairs(character:GetChildren()) do if item:IsA("Shirt") or item:IsA("Pants") or item:IsA("ShirtGraphic") then if not PlayerSkinOriginals.clothes[item] then PlayerSkinOriginals.clothes[item] = character end item.Parent = nil end end
    local skinMaterial = getMaterialByName(Config.player_skin_material)
    for _, item in ipairs(character:GetDescendants()) do
        if not item:FindFirstAncestorWhichIsA("Tool") then
            if item:IsA("BasePart") and item.Name ~= "HumanoidRootPart" then capturePlayerPart(item) item.Color, item.Material = Config.player_skin_color, skinMaterial end
            if item:IsA("MeshPart") or item:IsA("SpecialMesh") then capturePlayerMesh(item) pcall(function() if item:IsA("MeshPart") then item.TextureID = "" else item.TextureId = "" end end) end
        end
    end

local function updatePlayerChams(player, character)
    if not Config.enabled or not Config.chams then if chamsHighlights[player] then chamsHighlights[player]:Destroy() chamsHighlights[player] = nil end return end
    local hl = chamsHighlights[player] if not hl then hl = Instance.new("Highlight") hl.Parent = character chamsHighlights[player] = hl elseif hl.Parent ~= character then hl.Parent = character end
    local isVisible, chamColor, chamAlpha = checkVisibility(player, character), Config.chams_hidden_color, Config.chams_hidden_transparency
    if isVisible then chamColor, chamAlpha = Config.chams_visible_color, Config.chams_visible_transparency end
    hl.FillColor, hl.OutlineColor, hl.FillTransparency, hl.OutlineTransparency = chamColor, chamColor, chamAlpha, math.clamp(chamAlpha - 0.4, 0, 0.9)
end

local function updateBoxPositions(lines, x, y, w, h, thickness, color)
    for _, line in pairs(lines) do line.BackgroundColor3, line.Visible = color, true end
    if Config.box_style == "Full Box" then
        lines.tl_h.Position, lines.tl_h.Size, lines.tl_v.Visible = UDim2.new(0, x, 0, y), UDim2.new(0, w, 0, thickness), false
        lines.tr_h.Position, lines.tr_h.Size, lines.tr_v.Visible = UDim2.new(0, x, 0, y), UDim2.new(0, thickness, 0, h), false
        lines.bl_h.Position, lines.bl_h.Size, lines.bl_v.Visible = UDim2.new(0, x + w - thickness, 0, y), UDim2.new(0, thickness, 0, h), false
        lines.br_h.Position, lines.br_h.Size, lines.br_v.Visible = UDim2.new(0, x, 0, y + h - thickness), UDim2.new(0, w, 0, thickness), false
    else
        local len = math.clamp(w * 0.25, 5, 20) lines.tl_v.Visible, lines.tr_v.Visible, lines.bl_v.Visible, lines.br_v.Visible = true, true, true, true
        lines.tl_h.Position, lines.tl_h.Size = UDim2.new(0, x, 0, y), UDim2.new(0, len, 0, thickness)
        lines.tl_v.Position, lines.tl_v.Size = UDim2.new(0, x, 0, y), UDim2.new(0, thickness, 0, len)
        lines.tr_h.Position, lines.tr_h.Size = UDim2.new(0, x + w - len, 0, y), UDim2.new(0, len, 0, thickness)
        lines.tr_v.Position, lines.tr_v.Size = UDim2.new(0, x + w - thickness, 0, y), UDim2.new(0, thickness, 0, len)
        lines.bl_h.Position, lines.bl_h.Size = UDim2.new(0, x, 0, y + h - thickness), UDim2.new(0, len, 0, thickness)
        lines.bl_v.Position, lines.bl_v.Size = UDim2.new(0, x, 0, y + h - len), UDim2.new(0, thickness, 0, len)
        lines.br_h.Position, lines.br_h.Size = UDim2.new(0, x + w - len, 0, y + h - thickness), UDim2.new(0, len, 0, thickness)
        lines.br_v.Position, lines.br_v.Size = UDim2.new(0, x + w - thickness, 0, y + h - len), UDim2.new(0, thickness, 0, len)
    end
end

local function getDynamicBoundingBox(character)
    local minX, minY, maxX, maxY = math.huge, math.huge, -math.huge, -math.huge local anyOnScreen = false
    for _, part in ipairs(character:GetChildren()) do
        if part:IsA("BasePart") then
            local size, cframe = part.Size, part.CFrame
            local vertices = { cframe * Vector3.new(-size.X/2, -size.Y/2, -size.Z/2), cframe * Vector3.new( size.X/2, -size.Y/2, -size.Z/2), cframe * Vector3.new(-size.X/2, size.Y/2, -size.Z/2), cframe * Vector3.new(-size.X/2, -size.Y/2, size.Z/2), cframe * Vector3.new( size.X/2, size.Y/2, -size.Z/2), cframe * Vector3.new( size.X/2, -size.Y/2, size.Z/2), cframe * Vector3.new(-size.X/2, size.Y/2, size.Z/2), cframe * Vector3.new( size.X/2, size.Y/2, size.Z/2) }
            for i = 1, 8 do
                local screenPos, onScreen = Camera:WorldToViewportPoint(vertices[i])
                if onScreen then anyOnScreen = true end
                if screenPos.X < minX then minX = screenPos.X end if screenPos.X > maxX then maxX = screenPos.X end
                if screenPos.Y < minY then minY = screenPos.Y end if screenPos.Y > maxY then maxY = screenPos.Y end
            end
        end
    end
    if not anyOnScreen then return nil end
    local width, height = maxX - minX, maxY - minY
    if width < 5 or height < 5 then return nil end
    return minX, minY, width, height
end

local function processVisualsExecutionHookLoop()
    updateLighting() updateAtmosphere() updateWeaponSkins() updatePlayerSkin()
    if not Config.enabled then
        for player, data in pairs(espStorage) do
            data.container.Visible, data.nameLabel.Visible, data.healthLabel.Visible = false, false, false
            if chamsHighlights[player] then chamsHighlights[player]:Destroy() chamsHighlights[player] = nil end
        end return
    end
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            local data, character = espStorage[player], player.Character
            local humanoid = character and character:FindFirstChildOfClass("Humanoid")
            if character and humanoid and data then
                updatePlayerChams(player, character)
                local posX, posY, sizeX, sizeY = getDynamicBoundingBox(character)
                if posX then
                    local padding = 2 posX, posY, sizeX, sizeY = posX - padding, posY - padding, sizeX + (padding * 2), sizeY + (padding * 2)
                    if Config.box then
                        updateBoxPositions(data.lines, posX, posY, sizeX, sizeY, 1.5, Config.box_color) data.container.Visible = true
                        for _, line in pairs(data.lines) do line.BackgroundTransparency = Config.box_transparency local inline = line:FindFirstChildOfClass("Frame") if inline then inline.BackgroundTransparency = math.clamp(Config.box_transparency + 0.2, 0, 1) end end
                    else data.container.Visible = false end
                    if Config.names then
                        data.nameLabel.Text, data.nameLabel.TextColor3, data.nameLabel.TextTransparency, data.nameLabel.TextStrokeTransparency = player.DisplayName or player.Name, Config.names_color, Config.names_transparency, math.clamp(Config.names_transparency + 0.3, 0, 1)
                        data.nameLabel.Position, data.nameLabel.TextXAlignment, data.nameLabel.Visible = UDim2.new(0, posX + (sizeX / 2) - 100, 0, posY - 15), Enum.TextXAlignment.Center, true
                    else data.nameLabel.Visible = false end
                    if Config.health then
                        local hp, maxHp = math.max(0, math.floor(humanoid.Health)), humanoid.MaxHealth > 0 and humanoid.MaxHealth or 100
                        local pct = math.clamp(hp / maxHp, 0, 1) data.healthLabel.Text = hp .. " HP"
                        local chosenColor, chosenTransparency = Config.health_color, Config.health_transparency
                        if Config.health_dynamic then chosenColor, chosenTransparency = Config.health_low_color:Lerp(Config.health_high_color, pct), Config.health_low_transparency + (Config.health_high_transparency - Config.health_low_transparency) * pct end
                        data.healthLabel.TextColor3, data.healthLabel.TextTransparency, data.healthLabel.TextStrokeTransparency = chosenColor, chosenTransparency, math.clamp(chosenTransparency + 0.3, 0, 1)
                        data.healthLabel.Position, data.healthLabel.TextXAlignment, data.healthLabel.Visible = UDim2.new(0, posX + sizeX + 5, 0, posY + 1), Enum.TextXAlignment.Left, true
                    else data.healthLabel.Visible = false end
                else data.container.Visible, data.nameLabel.Visible, data.healthLabel.Visible = false, false, false end
            elseif data then data.container.Visible, data.nameLabel.Visible, data.healthLabel.Visible = false, false, false end
        end
    end
end

local applyTheme

local function updatePaletteColor(skipHexBoxUpdate)
    if not activeColorKey then return end
    local selectedColor = Color3.fromHSV(currentH, currentS, currentV)
    Config[activeColorKey], Config[activeAlphaKey] = selectedColor, currentAlpha
    alphaBgGradient.Color = ColorSequence.new({ ColorSequenceKeypoint.new(0, selectedColor), ColorSequenceKeypoint.new(1, selectedColor) })
    if activePreviewRect then activePreviewRect.BackgroundColor3, activePreviewRect.BackgroundTransparency = selectedColor, currentAlpha end
    if not skipHexBoxUpdate then hexBox.Text = colorToHex(selectedColor) end
    if activeColorKey == "menu_color" then applyTheme() elseif activeColorKey == "lighting_color" then updateLighting() elseif activeColorKey == "fog_color" then updateAtmosphere() elseif string.sub(activeColorKey, 1, 11) == "skin_color_" then updateWeaponSkins() elseif activeColorKey == "player_skin_color" then updatePlayerSkin() end
end

local function bindExternalColorPicker(colKey, alphaKey, previewRect, contextText)
    if activeColorKey == colKey then
        ColorPickerPanel.Visible = not ColorPickerPanel.Visible
        if not ColorPickerPanel.Visible then activeColorKey, activeAlphaKey, activePreviewRect = nil, nil, nil end return
    end
    activeColorKey, activeAlphaKey, activePreviewRect, CPIndicator.Text, ColorPickerPanel.Visible = colKey, alphaKey, previewRect, contextText and contextText:upper() or "SELECTING COLOR", true
    local col = Config[colKey] or Color3.fromRGB(255, 255, 255) currentAlpha = Config[alphaKey] or 0 currentH, currentS, currentV = Color3.toHSV(col)
    colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1) baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1))
    hueSlider.Position, pin.Position, alphaSlider.Position, hexBox.Text = UDim2.new(0, -2, currentH, 0), UDim2.new(currentS, 0, 1 - currentV, 0), UDim2.new(1 - currentAlpha, -1, 0, -2), colorToHex(col)
    alphaBgGradient.Color = ColorSequence.new({ColorSequenceKeypoint.new(0, col), ColorSequenceKeypoint.new(1, col)})
end

local dragCanvas, dragHue, dragAlpha = false, false, false

UserInputService.InputEnded:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then dragCanvas, dragHue, dragAlpha = false, false, false end end)
UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        if dragCanvas and ColorPickerPanel.Visible then local relativeX, relativeY = math.clamp((input.Position.X - canvas.AbsolutePosition.X) / canvas.AbsoluteSize.X, 0, 1), math.clamp((input.Position.Y - canvas.AbsolutePosition.Y) / canvas.AbsoluteSize.Y, 0, 1) currentS, currentV = relativeX, 1 - relativeY pin.Position = UDim2.new(relativeX, 0, relativeY, 0) updatePaletteColor(false)
        elseif dragHue and ColorPickerPanel.Visible then local relativeY = math.clamp((input.Position.Y - hueStrip.AbsolutePosition.Y) / hueStrip.AbsoluteSize.Y, 0, 1) currentH = relativeY hueSlider.Position = UDim2.new(0, -2, relativeY, 0) colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1) baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1)) updatePaletteColor(false)
        elseif dragAlpha and ColorPickerPanel.Visible then local relativeX = math.clamp((input.Position.X - alphaStrip.AbsolutePosition.X) / alphaStrip.AbsoluteSize.X, 0, 1) currentAlpha = 1 - relativeX alphaSlider.Position = UDim2.new(relativeX, -1, 0, -2) updatePaletteColor(false) end
    end
end)
