local fpsValue = 0
local fpsAccumulator = 0
local fpsFrames = 0
local hudAccumulator = 0

local function updateHUD(dt)
    fpsAccumulator = fpsAccumulator + dt
    fpsFrames = fpsFrames + 1
    hudAccumulator = hudAccumulator + dt

    if fpsAccumulator >= 0.5 then
        fpsValue = math.floor(fpsFrames / fpsAccumulator + 0.5)
        fpsAccumulator = 0
        fpsFrames = 0
    end
    
    if Config.array_list then
        ArrayListFrame.Visible = true
        for _, dictInfo in ipairs(aimFunctionsArrayKeys) do
            if Config[dictInfo.configValue] then
                if not activeArrayItems[dictInfo.name] then
                    local UIObj = generateArrayListItem(dictInfo.name)
                    UIObj.box.Parent = ArrayListFrame
                    UIObj.textLabel.TextColor3 = Config.menu_color
                    activeArrayItems[dictInfo.name] = UIObj
                else
                    activeArrayItems[dictInfo.name].textLabel.TextColor3 = Config.menu_color
                end
            else
                if activeArrayItems[dictInfo.name] then
                    activeArrayItems[dictInfo.name].box:Destroy()
                    activeArrayItems[dictInfo.name] = nil
                end
            end
        end
    else
        ArrayListFrame.Visible = false
        for idx, stored in pairs(activeArrayItems) do
            stored.box:Destroy()
        end
        table.clear(activeArrayItems)
    end

    WatermarkFrame.Visible = Config.watermark
    if not Config.watermark then return end

    WatermarkTitle.TextColor3 = Config.menu_color
    WatermarkStats.TextColor3 = Config.menu_color
    WatermarkAvatar.Visible = Config.watermark_pfp

    if hudAccumulator < 0.2 then return end
    hudAccumulator = 0

    local parts = {}
    if Config.watermark_fps then
        table.insert(parts, tostring(fpsValue) .. " FPS")
    end
    if Config.watermark_ping then
        local ping = 0
        pcall(function()
            ping = math.floor(LocalPlayer:GetNetworkPing() * 1000 + 0.5)
        end)
        table.insert(parts, tostring(ping) .. " MS")
    end
    if Config.watermark_name then
        table.insert(parts, LocalPlayer.DisplayName or LocalPlayer.Name)
    end

    WatermarkStats.Text = table.concat(parts, "  |  ")
    WatermarkStats.Visible = #parts > 0
    if #parts == 0 and not Config.watermark_pfp then
        WatermarkTitle.Size = UDim2.new(0, 76, 0, 18)
        WatermarkTitle.TextXAlignment = Enum.TextXAlignment.Center
    else
        WatermarkTitle.Size = UDim2.new(0, 64, 0, 18)
        WatermarkTitle.TextXAlignment = Enum.TextXAlignment.Left
    end
end

local function colorToHex(color)
    local r = math.round(color.R * 255)
    local g = math.round(color.G * 255)
    local b = math.round(color.B * 255)
    return string.format("#%02X%02X%02X", r, g, b)
end

local function hexToColor(hexStr)
    local success, color = pcall(function()
        return Color3.fromHex(hexStr)
    end)
    if success and color then
        return color
    end
    return nil
end

local function matchesBind(input, bind)
    if typeof(bind) == "EnumItem" then
        if bind.EnumType == Enum.KeyCode then
            return input.KeyCode == bind
        elseif bind.EnumType == Enum.UserInputType then
            return input.UserInputType == bind
        end
    end
    return false
end

local function getBindName(bind)
    if typeof(bind) == "EnumItem" then
        return bind.Name
    end
    return "None"
end

    if Config.fog then

        atmosphere.Haze = Config.fog_density

        end
    end
end

-- ==================== AIMBOT LOGIC ENGINE ====================

local function passesTeamCheck(targetPlayer)
    if not Config.team_check then return true end
    local tm = targetPlayer.Team
    if not tm then return true end
    
    local nm = string.lower(tm.Name)
    if Config.ignore_guards and (string.find(nm, "guard") or string.find(nm, "cop") or string.find(nm, "police")) then return false end
    if Config.ignore_inmates and (string.find(nm, "inmate") or string.find(nm, "prisoner")) then return false end
    if Config.ignore_criminals and (string.find(nm, "criminal")) then return false end
    
    return true
end

local function checkVisibility(player, character)
    if not Config.wall_check then return true end

    local root = character:FindFirstChild("HumanoidRootPart")
    if not root then return false end

    local camPos = Camera.CFrame.Position
    local targetPos = root.Position
    local direction = targetPos - camPos

    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    
    local filterTargets = {character}
    if LocalPlayer.Character then
        table.insert(filterTargets, LocalPlayer.Character)
    end
    params.FilterDescendantsInstances = filterTargets

    local result = workspace:Raycast(camPos, direction, params)
    return not result
end

local function getClosestTarget()
    local closestPlayer = nil
    local shortestDistance = math.huge
    local mousePos = UserInputService:GetMouseLocation()

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            if not passesTeamCheck(player) then continue end
            
            local character = player.Character
            local humanoid = character and character:FindFirstChildOfClass("Humanoid")
            if character and humanoid and humanoid.Health > 0 then
                local targetPart = character:FindFirstChild(Config.aimbot_target)
                if targetPart then
                    local screenPos, onScreen = Camera:WorldToViewportPoint(targetPart.Position)
                    if onScreen then
                        local distance = (Vector2.new(screenPos.X, screenPos.Y) - mousePos).Magnitude
                        if distance < Config.aimbot_radius and distance < shortestDistance then
                            if checkVisibility(player, character) then
                                shortestDistance = distance
                                closestPlayer = player
                            end
                        end
                    end
                end
            end
        end
    end
    return closestPlayer
end

local function aimAtTarget(targetPlayer)
    if not targetPlayer or not targetPlayer.Character then return end
    local targetPart = targetPlayer.Character:FindFirstChild(Config.aimbot_target)
    if targetPart then
        Camera.CFrame = CFrame.new(Camera.CFrame.Position, targetPart.Position)
    end
end

pcall(function()
    local oldNamecall
    oldNamecall = hookmetamethod(game, "__namecall", function(self, ...)
        if not checkcaller() and aimbotKeyPressed and Config.silent_aim then
            local method = getnamecallmethod()
            if method == "FireServer" and self.Name == "ShootEvent" then
                local tgtPlr = getClosestTarget()
                if tgtPlr and tgtPlr.Character then
                    local tgtPart = tgtPlr.Character:FindFirstChild(Config.aimbot_target)
                    if tgtPart then
                        local args = {...}
                        for i, v in ipairs(args) do
                            if typeof(v) == "Vector3" then args[i] = tgtPart.Position
                            elseif typeof(v) == "CFrame" then args[i] = tgtPart.CFrame
                            elseif typeof(v) == "Instance" and v:IsA("BasePart") then args[i] = tgtPart
                            elseif type(v) == "table" then
                                if v.Hit and typeof(v.Hit) == "Vector3" then v.Hit = tgtPart.Position end
                                if v.Part and typeof(v.Part) == "Instance" then v.Part = tgtPart end
                            end
                        end
                        return oldNamecall(self, unpack(args))
                    end
                end
            end
        end
        return oldNamecall(self, ...)
    end)
end)

local function processAimbot()
    local renderDrawVisual = (Config.aimbot or Config.silent_aim) and Config.draw_radius

    if renderDrawVisual then
        local mousePos = UserInputService:GetMouseLocation()
        FOVCircle.Position = UDim2.new(0, mousePos.X, 0, mousePos.Y)
        FOVCircle.Size = UDim2.new(0, Config.aimbot_radius * 2, 0, Config.aimbot_radius * 2)
        FOVCircle.Visible = true
        FOVStroke.Color = Config.draw_radius_color
        FOVStroke.Transparency = Config.draw_radius_transparency
    else
        FOVCircle.Visible = false
    end

    if Config.aimbot and not Config.silent_aim and aimbotKeyPressed then
        local target = getClosestTarget()
        if target then aimAtTarget(target) end
    end
end

local WeaponSkinNames = {

local WeaponSkinIds = {}
local WeaponSkinOriginals = {}
local WeaponSkinSelected = "AK47"

for _, weaponName in ipairs(WeaponSkinNames) do
    local id = weaponName:gsub("%W", "_")
    WeaponSkinIds[weaponName] = id
    Config["skin_enabled_" .. id] = false
    Config["skin_color_" .. id] = Color3.fromRGB(165, 90, 255)
    Config["skin_transparency_" .. id] = 0
    Config["skin_material_" .. id] = "Plastic"
    WeaponSkinOriginals[weaponName] = {}
end

Config.custom_weapon_skin = false

local function getMaterialByName(materialName)
    local name = tostring(materialName)
    local ok, material = pcall(function() return Enum.Material[name] end)
    if ok and material then return material end

    for _, m in ipairs(Enum.Material:GetEnumItems()) do
        if string.lower(m.Name) == string.lower(name) then return m end
    end

    return Enum.Material.Plastic
end

local function getWeaponRoots()
    local roots = {}
    local backpack = LocalPlayer:FindFirstChildOfClass("Backpack")
    if backpack then table.insert(roots, backpack) end
    if LocalPlayer.Character then table.insert(roots, LocalPlayer.Character) end
    return roots
end

local function namesMatch(instanceName, weaponName)
    return string.lower(instanceName) == string.lower(weaponName)
end

local function findWeaponInstances(weaponName)
    local found = {}
    for _, root in ipairs(getWeaponRoots()) do
        for _, item in ipairs(root:GetDescendants()) do
            if namesMatch(item.Name, weaponName) and (item:IsA("Tool") or item:IsA("Model")) then
                table.insert(found, item)
            end
        end
    end
    return found
end

local function captureOriginal(weaponName, meshObject, paintPart)
    local originals = WeaponSkinOriginals[weaponName]
    if originals[meshObject] then return end

    local textureProperty = meshObject:IsA("MeshPart") and "TextureID" or "TextureId"
    local textureValue = ""
    pcall(function() textureValue = meshObject[textureProperty] end)

    originals[meshObject] = {
        textureProperty = textureProperty, texture = textureValue, part = paintPart,
        color = paintPart and paintPart.Color or nil, material = paintPart and paintPart.Material or nil
    }
end

local function applySkinToMesh(weaponName, meshObject)
    local id = WeaponSkinIds[weaponName]
    local paintPart = meshObject:IsA("BasePart") and meshObject or meshObject.Parent
    if not paintPart or not paintPart:IsA("BasePart") then return end

    captureOriginal(weaponName, meshObject, paintPart)

    pcall(function()
        if meshObject:IsA("MeshPart") then meshObject.TextureID = "" else meshObject.TextureId = "" end
    end)

    paintPart.Color = Config["skin_color_" .. id]
    paintPart.Material = getMaterialByName(Config["skin_material_" .. id])
end

local function restoreWeaponSkin(weaponName)
    local originals = WeaponSkinOriginals[weaponName]
    if not originals then return end

    for meshObject, data in pairs(originals) do
        if meshObject and meshObject.Parent then
            pcall(function() meshObject[data.textureProperty] = data.texture or "" end)
            if data.part and data.part.Parent then
                if data.color then data.part.Color = data.color end
                if data.material then data.part.Material = data.material end
            end
        end
    end
    table.clear(originals)
end

local function restoreAllWeaponSkins()
    for _, weaponName in ipairs(WeaponSkinNames) do restoreWeaponSkin(weaponName) end
end

local function applyWeaponSkin(weaponName)
    local id = WeaponSkinIds[weaponName]
    if not Config.custom_weapon_skin or not Config["skin_enabled_" .. id] then
        restoreWeaponSkin(weaponName); return
    end

    for _, weapon in ipairs(findWeaponInstances(weaponName)) do
        for _, item in ipairs(weapon:GetDescendants()) do
            if item:IsA("MeshPart") or item:IsA("SpecialMesh") then applySkinToMesh(weaponName, item) end
        end
    end
end

local function updateWeaponSkins()
    if not Config.custom_weapon_skin then restoreAllWeaponSkins(); return end
    for _, weaponName in ipairs(WeaponSkinNames) do applyWeaponSkin(weaponName) end
end

local PlayerSkinOriginals = { clothes = {}, parts = {}, meshes = {}, decals = {} }

local function clearPlayerSkinCache()
    table.clear(PlayerSkinOriginals.clothes)
    table.clear(PlayerSkinOriginals.parts)
    table.clear(PlayerSkinOriginals.meshes)
    table.clear(PlayerSkinOriginals.decals)
end

local function restorePlayerSkin()
    for clothing, parent in pairs(PlayerSkinOriginals.clothes) do
        if clothing and not clothing.Parent and parent and parent.Parent then
            clothing.Parent = parent
        end
    end

    for part, data in pairs(PlayerSkinOriginals.parts) do
        if part and part.Parent then
            part.Color = data.color
            part.Material = data.material
        end

        local decalList = PlayerSkinOriginals.decals[part]
        if decalList then
            for _, info in ipairs(decalList) do
                if info.obj and info.obj.Parent then
                    info.obj.Transparency = info.transparency
                    pcall(function()
                        if info.obj:IsA("Decal") or info.obj:IsA("Texture") then info.obj.Texture = info.texture
                        elseif info.obj:IsA("SurfaceAppearance") then
                            info.obj.ColorMap = info.colorMap
                            info.obj.NormalMap = info.normalMap
                            info.obj.MetalnessMap = info.metalnessMap
                            info.obj.RoughnessMap = info.roughnessMap
                        end
                    end)
                end
            end
        end
    end

    for meshObject, data in pairs(PlayerSkinOriginals.meshes) do
        if meshObject and meshObject.Parent then
            pcall(function() meshObject[data.textureProperty] = data.texture or "" end)
        end
    end

    clearPlayerSkinCache()
end

local function capturePlayerPart(part)
    if PlayerSkinOriginals.parts[part] then return end
    PlayerSkinOriginals.parts[part] = { color = part.Color, material = part.Material }

    local decalList = {}
    for _, child in ipairs(part:GetChildren()) do
        if child:IsA("Decal") or child:IsA("Texture") or child:IsA("SurfaceAppearance") then
            local info = { obj = child, transparency = child.Transparency }

            if child:IsA("Decal") or child:IsA("Texture") then info.texture = child.Texture
            elseif child:IsA("SurfaceAppearance") then
                info.colorMap = child.ColorMap
                info.normalMap = child.NormalMap
                info.metalnessMap = child.MetalnessMap
                info.roughnessMap = child.RoughnessMap
            end

            table.insert(decalList, info)
            child.Transparency = 1
            pcall(function()
                if child:IsA("Decal") or child:IsA("Texture") then child.Texture = ""
                elseif child:IsA("SurfaceAppearance") then
                    child.ColorMap = ""
                    child.NormalMap = ""
                    child.MetalnessMap = ""
                    child.RoughnessMap = ""
                end
            end)
        end
    end

    if next(decalList) then PlayerSkinOriginals.decals[part] = decalList end
end

local function capturePlayerMesh(meshObject)
    if PlayerSkinOriginals.meshes[meshObject] then return end

    local textureProperty
    if meshObject:IsA("MeshPart") then textureProperty = "TextureID"
    elseif meshObject:IsA("SpecialMesh") then textureProperty = "TextureId"
    else return end

    local textureValue = ""
    pcall(function() textureValue = meshObject[textureProperty] end)
    PlayerSkinOriginals.meshes[meshObject] = { textureProperty = textureProperty, texture = textureValue }
end

local function applyPlayerSkin()
    local character = LocalPlayer.Character
    if not character then return end

    for _, item in ipairs(character:GetChildren()) do
        if item:IsA("Shirt") or item:IsA("Pants") or item:IsA("ShirtGraphic") then
            if not PlayerSkinOriginals.clothes[item] then PlayerSkinOriginals.clothes[item] = character end
            item.Parent = nil
        end
    end

    local skinMaterial = getMaterialByName(Config.player_skin_material)
    for _, item in ipairs(character:GetDescendants()) do
        if item:FindFirstAncestorWhichIsA("Tool") then
            -- skip weapon/tool parts
        elseif item:IsA("BasePart") and item.Name ~= "HumanoidRootPart" then
            capturePlayerPart(item)
            item.Color = Config.player_skin_color
            item.Material = skinMaterial
        end

        if item:IsA("MeshPart") or item:IsA("SpecialMesh") then
            if not item:FindFirstAncestorWhichIsA("Tool") then
                capturePlayerMesh(item)
                pcall(function()
                    if item:IsA("MeshPart") then item.TextureID = "" else item.TextureId = "" end
                end)
            end
        end
    end
end

local function updatePlayerSkin()
    if Config.custom_player_skin then applyPlayerSkin() else restorePlayerSkin() end
end

local function updatePlayerChams(player, character)
    if not Config.enabled or not Config.chams then
        if chamsHighlights[player] then
            chamsHighlights[player]:Destroy()
            chamsHighlights[player] = nil
        end
        return
    end

    local hl = chamsHighlights[player]
    if not hl then
        hl = Instance.new("Highlight")
        hl.Parent = character
        chamsHighlights[player] = hl
    elseif hl.Parent ~= character then
        hl.Parent = character
    end

    local isVisible = checkVisibility(player, character)
    local chamColor = isVisible and Config.chams_visible_color or Config.chams_hidden_color
    local chamAlpha = isVisible and Config.chams_visible_transparency or Config.chams_hidden_transparency

    hl.FillColor = chamColor
    hl.OutlineColor = chamColor
    hl.FillTransparency = chamAlpha
    hl.OutlineTransparency = math.clamp(chamAlpha - 0.4, 0, 0.9)
end

local function updateBoxPositions(lines, x, y, w, h, thickness, color)
    for _, line in pairs(lines) do line.BackgroundColor3 = color; line.Visible = true end

    if Config.box_style == "Full Box" then
        lines.tl_h.Position = UDim2.new(0, x, 0, y); lines.tl_h.Size = UDim2.new(0, w, 0, thickness); lines.tl_v.Visible = false
        lines.tr_h.Position = UDim2.new(0, x, 0, y); lines.tr_h.Size = UDim2.new(0, thickness, 0, h); lines.tr_v.Visible = false
        lines.bl_h.Position = UDim2.new(0, x + w - thickness, 0, y); lines.bl_h.Size = UDim2.new(0, thickness, 0, h); lines.bl_v.Visible = false
        lines.br_h.Position = UDim2.new(0, x, 0, y + h - thickness); lines.br_h.Size = UDim2.new(0, w, 0, thickness); lines.br_v.Visible = false
    else
        local len = math.clamp(w * 0.25, 5, 20)
        lines.tl_v.Visible = true; lines.tr_v.Visible = true; lines.bl_v.Visible = true; lines.br_v.Visible = true
        lines.tl_h.Position = UDim2.new(0, x, 0, y); lines.tl_h.Size = UDim2.new(0, len, 0, thickness)
        lines.tl_v.Position = UDim2.new(0, x, 0, y); lines.tl_v.Size = UDim2.new(0, thickness, 0, len)
        lines.tr_h.Position = UDim2.new(0, x + w - len, 0, y); lines.tr_h.Size = UDim2.new(0, len, 0, thickness)
        lines.tr_v.Position = UDim2.new(0, x + w - thickness, 0, y); lines.tr_v.Size = UDim2.new(0, thickness, 0, len)
        lines.bl_h.Position = UDim2.new(0, x, 0, y + h - thickness); lines.bl_h.Size = UDim2.new(0, len, 0, thickness)
        lines.bl_v.Position = UDim2.new(0, x, 0, y + h - len); lines.bl_v.Size = UDim2.new(0, thickness, 0, len)
        lines.br_h.Position = UDim2.new(0, x + w - len, 0, y + h - thickness); lines.br_h.Size = UDim2.new(0, len, 0, thickness)
        lines.br_v.Position = UDim2.new(0, x + w - thickness, 0, y + h - len); lines.br_v.Size = UDim2.new(0, thickness, 0, len)
    end
end

local function getDynamicBoundingBox(character)
    local minX, minY = math.huge, math.huge
    local maxX, maxY = -math.huge, -math.huge
    local anyOnScreen = false

    for _, part in ipairs(character:GetChildren()) do
        if part:IsA("BasePart") then
            local size = part.Size; local cframe = part.CFrame

            local vertices = {
                cframe * Vector3.new(-size.X/2, -size.Y/2, -size.Z/2), cframe * Vector3.new( size.X/2, -size.Y/2, -size.Z/2),
                cframe * Vector3.new(-size.X/2,  size.Y/2, -size.Z/2), cframe * Vector3.new(-size.X/2, -size.Y/2,  size.Z/2),
                cframe * Vector3.new( size.X/2,  size.Y/2, -size.Z/2), cframe * Vector3.new( size.X/2, -size.Y/2,  size.Z/2),
                cframe * Vector3.new(-size.X/2,  size.Y/2,  size.Z/2), cframe * Vector3.new( size.X/2,  size.Y/2,  size.Z/2)
            }

            for i = 1, 8 do
                local screenPos, onScreen = Camera:WorldToViewportPoint(vertices[i])
                if onScreen then anyOnScreen = true end
                if screenPos.X < minX then minX = screenPos.X end; if screenPos.X > maxX then maxX = screenPos.X end
                if screenPos.Y < minY then minY = screenPos.Y end; if screenPos.Y > maxY then maxY = screenPos.Y end
            end
        end
    end

    if not anyOnScreen then return nil end

    local width = maxX - minX; local height = maxY - minY
    if width < 5 or height < 5 then return nil end
    return minX, minY, width, height
end

local function processVisuals()
    updateLighting()
    updateAtmosphere()
    updateWeaponSkins()
    updatePlayerSkin()

    if not Config.enabled then
        for player, _ in pairs(espStorage) do
            local data = espStorage[player]
            data.container.Visible = false; data.nameLabel.Visible = false; data.healthLabel.Visible = false
            if chamsHighlights[player] then chamsHighlights[player]:Destroy(); chamsHighlights[player] = nil end
        end
        return
    end

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            local data = espStorage[player]
            local character = player.Character
            local humanoid = character and character:FindFirstChildOfClass("Humanoid")

            if character and humanoid and data then
                updatePlayerChams(player, character)

                local posX, posY, sizeX, sizeY = getDynamicBoundingBox(character)

                if posX then
                    local padding = 2
                    posX = posX - padding; posY = posY - padding
                    sizeX = sizeX + (padding * 2); sizeY = sizeY + (padding * 2)

                    if Config.box then
                        updateBoxPositions(data.lines, posX, posY, sizeX, sizeY, 1.5, Config.box_color)
                        data.container.Visible = true
                        for _, line in pairs(data.lines) do
                            line.BackgroundTransparency = Config.box_transparency
                            local inline = line:FindFirstChildOfClass("Frame")
                            if inline then inline.BackgroundTransparency = math.clamp(Config.box_transparency + 0.2, 0, 1) end
                        end
                    else
                        data.container.Visible = false
                    end

                    if Config.names then
                        data.nameLabel.Text = player.DisplayName or player.Name
                        data.nameLabel.TextColor3 = Config.names_color
                        data.nameLabel.TextTransparency = Config.names_transparency
                        data.nameLabel.TextStrokeTransparency = math.clamp(Config.names_transparency + 0.3, 0, 1)
                        data.nameLabel.Position = UDim2.new(0, posX + (sizeX / 2) - 100, 0, posY - 15)
                        data.nameLabel.TextXAlignment = Enum.TextXAlignment.Center; data.nameLabel.Visible = true
                    else data.nameLabel.Visible = false end

                    if Config.health then
                        local hp = math.max(0, math.floor(humanoid.Health))
                        local maxHp = humanoid.MaxHealth > 0 and humanoid.MaxHealth or 100
                        local pct = math.clamp(hp / maxHp, 0, 1)

                        data.healthLabel.Text = hp .. " HP"

                        local chosenColor, chosenTransparency
                        if Config.health_dynamic then
                            chosenColor = Config.health_low_color:Lerp(Config.health_high_color, pct)
                            chosenTransparency = Config.health_low_transparency + (Config.health_high_transparency - Config.health_low_transparency) * pct
                        else
                            chosenColor = Config.health_color
                            chosenTransparency = Config.health_transparency
                        end

                        data.healthLabel.TextColor3 = chosenColor
                        data.healthLabel.TextTransparency = chosenTransparency
                        data.healthLabel.TextStrokeTransparency = math.clamp(chosenTransparency + 0.3, 0, 1)
                        data.healthLabel.Position = UDim2.new(0, posX + sizeX + 5, 0, posY + 1)
                        data.healthLabel.TextXAlignment = Enum.TextXAlignment.Left; data.healthLabel.Visible = true
                    else data.healthLabel.Visible = false end
                else
                    data.container.Visible = false; data.nameLabel.Visible = false; data.healthLabel.Visible = false
                end
            else
                if data then data.container.Visible = false; data.nameLabel.Visible = false; data.healthLabel.Visible = false end
            end
        end
    end
end

local function createCollapsibleSection(title, parent)
    local sectionFrame = Instance.new("Frame")
    sectionFrame.Size = UDim2.new(1, 0, 0, 26); sectionFrame.BackgroundTransparency = 1; sectionFrame.Parent = parent

    local sectionLayout = Instance.new("UIListLayout")
    sectionLayout.SortOrder = Enum.SortOrder.LayoutOrder; sectionLayout.Padding = UDim.new(0, 6); sectionLayout.Parent = sectionFrame

    local header = Instance.new("TextButton")
    header.Size = UDim2.new(1, 0, 0, 26); header.BackgroundColor3 = Color3.fromRGB(32, 32, 32); header.BorderSizePixel = 0
    header.Text = "  [+] " .. title:upper(); header.TextColor3 = Config.menu_color; header.Font = Enum.Font.GothamBold
    header.TextSize = 10; header.TextXAlignment = Enum.TextXAlignment.Left; header.Parent = sectionFrame

    local headerStroke = Instance.new("UIStroke")
    headerStroke.Thickness = 1; headerStroke.Color = Color3.fromRGB(52, 52, 52); headerStroke.Parent = header
    table.insert(collapsibleHeaders, header)

    local contentFrame = Instance.new("Frame")
    contentFrame.Size = UDim2.new(1, 0, 0, 0); contentFrame.BackgroundTransparency = 1; contentFrame.Visible = false; contentFrame.Parent = sectionFrame

    local contentLayout = Instance.new("UIListLayout")
    contentLayout.SortOrder = Enum.SortOrder.LayoutOrder; contentLayout.Padding = UDim.new(0, 6); contentLayout.Parent = contentFrame

    local collapsed = true
    contentLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
        if not collapsed then
            contentFrame.Size = UDim2.new(1, 0, 0, contentLayout.AbsoluteContentSize.Y)
            sectionFrame.Size = UDim2.new(1, 0, 0, 26 + 6 + contentLayout.AbsoluteContentSize.Y)
        end
    end)

    header.MouseButton1Click:Connect(function()
        collapsed = not collapsed; contentFrame.Visible = not collapsed
        if collapsed then
            header.Text = "  [+] " .. title:upper(); contentFrame.Size = UDim2.new(1, 0, 0, 0); sectionFrame.Size = UDim2.new(1, 0, 0, 26)
        else
            header.Text = "  [-] " .. title:upper(); contentFrame.Size = UDim2.new(1, 0, 0, contentLayout.AbsoluteContentSize.Y)
            sectionFrame.Size = UDim2.new(1, 0, 0, 26 + 6 + contentLayout.AbsoluteContentSize.Y)
        end
    end)

    return contentFrame
end

local activeColorKey = nil; local activeAlphaKey = nil; local activePreviewRect = nil
local currentH, currentS, currentV, currentAlpha = 0, 0, 0, 0
local applyTheme

local function updatePaletteColor(skipHexBoxUpdate)
    if not activeColorKey then return end
    local selectedColor = Color3.fromHSV(currentH, currentS, currentV)
    Config[activeColorKey] = selectedColor
    Config[activeAlphaKey] = currentAlpha

    alphaBgGradient.Color = ColorSequence.new({ColorSequenceKeypoint.new(0, selectedColor), ColorSequenceKeypoint.new(1, selectedColor)})
    if activePreviewRect then activePreviewRect.BackgroundColor3 = selectedColor; activePreviewRect.BackgroundTransparency = currentAlpha end
    if not skipHexBoxUpdate then hexBox.Text = colorToHex(selectedColor) end
    
    if activeColorKey == "menu_color" then applyTheme()
    elseif activeColorKey == "lighting_color" then updateLighting()
    elseif activeColorKey == "fog_color" then updateAtmosphere()
    elseif string.sub(activeColorKey, 1, 11) == "skin_color_" then updateWeaponSkins()
    elseif activeColorKey == "player_skin_color" then updatePlayerSkin() end
end

local function bindExternalColorPicker(colKey, alphaKey, previewRect, contextText)
    if activeColorKey == colKey then
        ColorPickerPanel.Visible = not ColorPickerPanel.Visible
        if not ColorPickerPanel.Visible then activeColorKey = nil; activeAlphaKey = nil; activePreviewRect = nil end return
    end
    activeColorKey = colKey; activeAlphaKey = alphaKey; activePreviewRect = previewRect; CPIndicator.Text = contextText and contextText:upper() or "SELECTING COLOR"; ColorPickerPanel.Visible = true

    local col = Config[colKey] or Color3.fromRGB(255, 255, 255)
    currentAlpha = Config[alphaKey] or 0; currentH, currentS, currentV = Color3.toHSV(col)

    colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1); baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1))
    hueSlider.Position = UDim2.new(0, -2, currentH, 0); pin.Position = UDim2.new(currentS, 0, 1 - currentV, 0); alphaSlider.Position = UDim2.new(1 - currentAlpha, -1, 0, -2)
    hexBox.Text = colorToHex(col)
    alphaBgGradient.Color = ColorSequence.new({ColorSequenceKeypoint.new(0, col), ColorSequenceKeypoint.new(1, col)})
end

    if not activeColorKey then return end

end)

    if not activeColorKey then return end

end)
UserInputService.InputEnded:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then dragCanvas = false; dragHue = false; dragAlpha = false end end)
UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        if dragCanvas and ColorPickerPanel.Visible then local relativeX = math.clamp((input.Position.X - canvas.AbsolutePosition.X) / canvas.AbsoluteSize.X, 0, 1); local relativeY = math.clamp((input.Position.Y - canvas.AbsolutePosition.Y) / canvas.AbsoluteSize.Y, 0, 1); currentS = relativeX; currentV = 1 - relativeY; pin.Position = UDim2.new(relativeX, 0, relativeY, 0); updatePaletteColor(false)
        elseif dragHue and ColorPickerPanel.Visible then local relativeY = math.clamp((input.Position.Y - hueStrip.AbsolutePosition.Y) / hueStrip.AbsoluteSize.Y, 0, 1); currentH = relativeY; hueSlider.Position = UDim2.new(0, -2, relativeY, 0); colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1); baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1)); updatePaletteColor(false)
        elseif dragAlpha and ColorPickerPanel.Visible then local relativeX = math.clamp((input.Position.X - alphaStrip.AbsolutePosition.X) / alphaStrip.AbsoluteSize.X, 0, 1); currentAlpha = 1 - relativeX; alphaSlider.Position = UDim2.new(relativeX, -1, 0, -2); updatePaletteColor(false) end
    end
end)

-- ==================== SYSTEM TRANSITION ANIMATION ====================

local transitionActive = false

local function toggleMenu()
    if transitionActive then return end
    transitionActive = true
    menuVisible = not menuVisible

    if menuVisible then
        MainFrame.Visible = true
        TweenService:Create(MainFrame, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        TweenService:Create(Sidebar, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        TweenService:Create(Logo, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {TextTransparency = 0}):Play()
        TweenService:Create(MainStroke, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0}):Play()
        TweenService:Create(SepLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        TweenService:Create(AccentLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        task.wait(0.15)
    else
        TweenService:Create(MainFrame, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        TweenService:Create(Sidebar, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        TweenService:Create(Logo, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {TextTransparency = 1}):Play()
        TweenService:Create(MainStroke, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {Transparency = 1}):Play()
        TweenService:Create(SepLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        TweenService:Create(AccentLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        task.wait(0.15)
        if not menuVisible then
            MainFrame.Visible = false
            ColorPickerPanel.Visible = false
            activeColorKey = nil; activeAlphaKey = nil; activePreviewRect = nil
        end
    end
    transitionActive = false
end

table.insert(connections, RunService.RenderStepped:Connect(function(dt)
    processVisuals()
    processAimbot()
    updateHUD(dt)
end))

table.insert(connections, UserInputService.InputBegan:Connect(function(input, processed)
    if UIRefs.listeningForBind and input.UserInputType == Enum.UserInputType.Keyboard then
        Config.toggleKey = input.KeyCode
        UIRefs.BindButton.Text = input.KeyCode.Name
        UIRefs.BindButton.TextColor3 = Config.menu_color
        UIRefs.listeningForBind = false
        return
    end

    if listeningForAimbotBind then
        local bound = nil
        if input.UserInputType == Enum.UserInputType.Keyboard and input.KeyCode ~= Enum.KeyCode.Unknown then
            bound = input.KeyCode
        elseif input.UserInputType == Enum.UserInputType.MouseButton1 or
               input.UserInputType == Enum.UserInputType.MouseButton2 or
               input.UserInputType == Enum.UserInputType.MouseButton3 then
            bound = input.UserInputType
        end

        if bound then
            Config.aimbot_key = bound
            UIRefs.AimbotBindBtn.Text = getBindName(bound)
            UIRefs.AimbotBindBtn.TextColor3 = Config.menu_color
            listeningForAimbotBind = false
            aimbotKeyPressed = false
            return
        end
    end

    if processed then return end

    if input.KeyCode == Config.toggleKey then
        toggleMenu()
    end

    if matchesBind(input, Config.aimbot_key) then
        if Config.aimbot_mode == "Hold" then
            aimbotKeyPressed = true
        elseif Config.aimbot_mode == "Toggle" then
            aimbotKeyPressed = not aimbotKeyPressed
        end
    end
end))

table.insert(connections, UserInputService.InputEnded:Connect(function(input, processed)
    if matchesBind(input, Config.aimbot_key) then
        if Config.aimbot_mode == "Hold" then
            aimbotKeyPressed = false
        end
    end
end))
