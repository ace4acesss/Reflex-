local fpsValue = 0
local fpsAccumulator = 0
local fpsFrames = 0
local watermarkAccumulator = 0

local function updateWatermark(dt)
    fpsAccumulator = fpsAccumulator + dt
    fpsFrames = fpsFrames + 1
    watermarkAccumulator = watermarkAccumulator + dt

    if fpsAccumulator >= 0.5 then
        fpsValue = math.floor(fpsFrames / fpsAccumulator + 0.5)
        fpsAccumulator = 0
        fpsFrames = 0
    end

    WatermarkFrame.Visible = Config.watermark
    if not Config.watermark then return end

    WatermarkTitle.TextColor3 = Config.menu_color
    WatermarkStats.TextColor3 = Config.menu_color
    WatermarkAvatar.Visible = Config.watermark_pfp

    if watermarkAccumulator < 0.2 then return end
    watermarkAccumulator = 0

    local parts = {}
    if Config.watermark_fps then
        table.insert(parts, tostring(fpsValue) .. " FPS")
    end
    if Config.watermark_ping then
        local ping = 0
        pcall(function()
            ping = math.floor(LocalPlayer:GetNetworkPing() * 1000 + 0.5)
        end)
        table.insert(parts, tostring(ping) .. " MS")
    end
    if Config.watermark_name then
        table.insert(parts, LocalPlayer.DisplayName or LocalPlayer.Name)
    end

    WatermarkStats.Text = table.concat(parts, "  |  ")
    WatermarkStats.Visible = #parts > 0
    if #parts == 0 and not Config.watermark_pfp then
        WatermarkTitle.Size = UDim2.new(0, 76, 0, 18)
        WatermarkTitle.TextXAlignment = Enum.TextXAlignment.Center
    else
        WatermarkTitle.Size = UDim2.new(0, 64, 0, 18)
        WatermarkTitle.TextXAlignment = Enum.TextXAlignment.Left
    end
end

-- ==================== CORE CONVERSION HELPERS ====================

local function colorToHex(color)
    local r = math.round(color.R * 255)
    local g = math.round(color.G * 255)
    local b = math.round(color.B * 255)
    return string.format("#%02X%02X%02X", r, g, b)
end

local function hexToColor(hexStr)
    local success, color = pcall(function()
        return Color3.fromHex(hexStr)
    end)
    if success and color then
        return color
    end
    return nil
end

local function getAtmosphere()
    local atmosphere = Lighting:FindFirstChildOfClass("Atmosphere")
    if not atmosphere and Config.fog then
        atmosphere = Instance.new("Atmosphere")
        atmosphere.Name = "ReflexAtmosphere"
        atmosphere.Parent = Lighting
        createdAtmosphere = atmosphere
    end
    return atmosphere
end

local function updateAtmosphere()
    local atmosphere = getAtmosphere()
    if not atmosphere then return end

    if Config.fog then
        atmosphere.Density = Config.fog_density
        atmosphere.Haze = Config.fog_density
        atmosphere.Offset = 1
        atmosphere.Color = Config.fog_color
        atmosphere.Decay = Config.fog_color
    else
        atmosphere.Density = 0
        atmosphere.Haze = 0
        if createdAtmosphere == atmosphere then
            atmosphere:Destroy()
            createdAtmosphere = nil
        end
    end
end

local WeaponSkinIds = {}
local WeaponSkinOriginals = {}
local WeaponSkinSelected = "AK47"

for _, weaponName in ipairs(WeaponSkinNames) do
    local id = weaponName:gsub("%W", "_")
    WeaponSkinIds[weaponName] = id
    Config["skin_enabled_" .. id] = false
    Config["skin_color_" .. id] = Color3.fromRGB(165, 90, 255)
    Config["skin_transparency_" .. id] = 0
    Config["skin_material_" .. id] = "Plastic"
    WeaponSkinOriginals[weaponName] = {}
end

Config.custom_weapon_skin = false

local function getMaterialByName(materialName)
    local ok, material = pcall(function()
        return Enum.Material[materialName]
    end)
    if ok and material then
        return material
    end
    return Enum.Material.Plastic
end

local function getWeaponRoots()
    local roots = {}
    local backpack = LocalPlayer:FindFirstChildOfClass("Backpack")
    if backpack then
        table.insert(roots, backpack)
    end
    if LocalPlayer.Character then
        table.insert(roots, LocalPlayer.Character)
    end
    return roots
end

local function namesMatch(instanceName, weaponName)
    return string.lower(instanceName) == string.lower(weaponName)
end

local function findWeaponInstances(weaponName)
    local found = {}
    for _, root in ipairs(getWeaponRoots()) do
        for _, item in ipairs(root:GetDescendants()) do
            if namesMatch(item.Name, weaponName) and (item:IsA("Tool") or item:IsA("Model")) then
                table.insert(found, item)
            end
        end
    end
    return found
end

local function captureOriginal(weaponName, meshObject, paintPart)
    local originals = WeaponSkinOriginals[weaponName]
    if originals[meshObject] then return end

    local textureProperty = meshObject:IsA("MeshPart") and "TextureID" or "TextureId"
    local textureValue = ""
    pcall(function()
        textureValue = meshObject[textureProperty]
    end)

    originals[meshObject] = {
        textureProperty = textureProperty,
        texture = textureValue,
        part = paintPart,
        color = paintPart and paintPart.Color or nil,
        material = paintPart and paintPart.Material or nil
    }
end

local function applySkinToMesh(weaponName, meshObject)
    local id = WeaponSkinIds[weaponName]
    local paintPart = meshObject:IsA("BasePart") and meshObject or meshObject.Parent
    if not paintPart or not paintPart:IsA("BasePart") then return end

    captureOriginal(weaponName, meshObject, paintPart)

    pcall(function()
        if meshObject:IsA("MeshPart") then
            meshObject.TextureID = ""
        else
            meshObject.TextureId = ""
        end
    end)

    paintPart.Color = Config["skin_color_" .. id]
    paintPart.Material = getMaterialByName(Config["skin_material_" .. id])
end

local function restoreWeaponSkin(weaponName)
    local originals = WeaponSkinOriginals[weaponName]
    if not originals then return end

    for meshObject, data in pairs(originals) do
        if meshObject and meshObject.Parent then
            pcall(function()
                meshObject[data.textureProperty] = data.texture or ""
            end)
            if data.part and data.part.Parent then
                if data.color then data.part.Color = data.color end
                if data.material then data.part.Material = data.material end
            end
        end
    end
    table.clear(originals)
end

local function restoreAllWeaponSkins()
    for _, weaponName in ipairs(WeaponSkinNames) do
        restoreWeaponSkin(weaponName)
    end
end

local function applyWeaponSkin(weaponName)
    local id = WeaponSkinIds[weaponName]
    if not Config.custom_weapon_skin or not Config["skin_enabled_" .. id] then
        restoreWeaponSkin(weaponName)
        return
    end

    for _, weapon in ipairs(findWeaponInstances(weaponName)) do
        for _, item in ipairs(weapon:GetDescendants()) do
            if item:IsA("MeshPart") or item:IsA("SpecialMesh") then
                applySkinToMesh(weaponName, item)
            end
        end
    end
end

local function updateWeaponSkins()
    if not Config.custom_weapon_skin then
        restoreAllWeaponSkins()
        return
    end

    for _, weaponName in ipairs(WeaponSkinNames) do
        applyWeaponSkin(weaponName)
    end
end

Config.custom_player_skin = false
Config.player_skin_color = Color3.fromRGB(165, 90, 255)
Config.player_skin_transparency = 0
Config.player_skin_material = "Plastic"

local PlayerSkinOriginals = {
    clothes = {},
    parts = {},
    meshes = {}
}

local function clearPlayerSkinCache()
    table.clear(PlayerSkinOriginals.clothes)
    table.clear(PlayerSkinOriginals.parts)
    table.clear(PlayerSkinOriginals.meshes)
end

local function restorePlayerSkin()
    for clothing, parent in pairs(PlayerSkinOriginals.clothes) do
        if clothing and not clothing.Parent and parent and parent.Parent then
            clothing.Parent = parent
        end
    end

    for part, data in pairs(PlayerSkinOriginals.parts) do
        if part and part.Parent then
            part.Color = data.color
            part.Material = data.material
        end
    end

    for meshObject, data in pairs(PlayerSkinOriginals.meshes) do
        if meshObject and meshObject.Parent then
            pcall(function()
                meshObject[data.textureProperty] = data.texture or ""
            end)
        end
    end

    clearPlayerSkinCache()
end

local function capturePlayerPart(part)
    if PlayerSkinOriginals.parts[part] then return end
    PlayerSkinOriginals.parts[part] = {
        color = part.Color,
        material = part.Material
    }
end

local function capturePlayerMesh(meshObject)
    if PlayerSkinOriginals.meshes[meshObject] then return end
    local textureProperty = meshObject:IsA("MeshPart") and "TextureID" or "TextureId"
    local textureValue = ""
    pcall(function()
        textureValue = meshObject[textureProperty]
    end)
    PlayerSkinOriginals.meshes[meshObject] = {
        textureProperty = textureProperty,
        texture = textureValue
    }
end

local function applyPlayerSkin()
    local character = LocalPlayer.Character
    if not character then return end

    for _, item in ipairs(character:GetChildren()) do
        if item:IsA("Shirt") or item:IsA("Pants") or item:IsA("ShirtGraphic") then
            if not PlayerSkinOriginals.clothes[item] then
                PlayerSkinOriginals.clothes[item] = character
            end
            item.Parent = nil
        end
    end

    local skinMaterial = getMaterialByName(Config.player_skin_material)
    for _, item in ipairs(character:GetDescendants()) do
        if item:IsA("BasePart") and item.Name ~= "HumanoidRootPart" then
            capturePlayerPart(item)
            item.Color = Config.player_skin_color
            item.Material = skinMaterial
        end

        if item:IsA("MeshPart") or item:IsA("SpecialMesh") then
            capturePlayerMesh(item)
            pcall(function()
                if item:IsA("MeshPart") then
                    item.TextureID = ""
                else
                    item.TextureId = ""
                end
            end)
        end
    end
end

local function updatePlayerSkin()
    if Config.custom_player_skin then
        applyPlayerSkin()
    else
        restorePlayerSkin()
    end
end

local function checkVisibility(player, character)
    local root = character:FindFirstChild("HumanoidRootPart")
    if not root then return false end

    local camPos = Camera.CFrame.Position
    local targetPos = root.Position
    local direction = targetPos - camPos

    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    
    local filterTargets = {character}
    if LocalPlayer.Character then
        table.insert(filterTargets, LocalPlayer.Character)
    end
    params.FilterDescendantsInstances = filterTargets

    local result = workspace:Raycast(camPos, direction, params)
    return not result
end

local function removeDrawings(player)
    local data = espStorage[player]
    if data then
        data.container:Destroy()
        data.nameLabel:Destroy()
        data.healthLabel:Destroy()
        espStorage[player] = nil
    end
    if chamsHighlights[player] then
        chamsHighlights[player]:Destroy()
        chamsHighlights[player] = nil
    end
end

local function updatePlayerChams(player, character)
    if not Config.enabled or not Config.chams then
        if chamsHighlights[player] then
            chamsHighlights[player]:Destroy()
            chamsHighlights[player] = nil
        end
        return
    end

    local hl = chamsHighlights[player]
    if not hl then
        hl = Instance.new("Highlight")
        hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
        hl.Parent = character
        chamsHighlights[player] = hl
    elseif hl.Parent ~= character then
        hl.Parent = character
    end

    local isVisible = checkVisibility(player, character)
    local chamColor = isVisible and Config.chams_visible_color or Config.chams_hidden_color
    local chamAlpha = isVisible and Config.chams_visible_transparency or Config.chams_hidden_transparency

    hl.FillColor = chamColor
    hl.OutlineColor = chamColor
    hl.FillTransparency = chamAlpha
    hl.OutlineTransparency = math.clamp(chamAlpha - 0.4, 0, 0.9)
end

local function updateBoxPositions(lines, x, y, w, h, thickness, color)
    for _, line in pairs(lines) do
        line.BackgroundColor3 = color
        line.Visible = true
    end

    if Config.box_style == "Full Box" then
        lines.tl_h.Position = UDim2.new(0, x, 0, y)
        lines.tl_h.Size = UDim2.new(0, w, 0, thickness)
        lines.tl_v.Visible = false

        lines.tr_h.Position = UDim2.new(0, x, 0, y)
        lines.tr_h.Size = UDim2.new(0, thickness, 0, h)
        lines.tr_v.Visible = false

        lines.bl_h.Position = UDim2.new(0, x + w - thickness, 0, y)
        lines.bl_h.Size = UDim2.new(0, thickness, 0, h)
        lines.bl_v.Visible = false

        lines.br_h.Position = UDim2.new(0, x, 0, y + h - thickness)
        lines.br_h.Size = UDim2.new(0, w, 0, thickness)
        lines.br_v.Visible = false
    else
        local len = math.clamp(w * 0.25, 5, 20)
        lines.tl_v.Visible = true; lines.tr_v.Visible = true
        lines.bl_v.Visible = true; lines.br_v.Visible = true

        lines.tl_h.Position = UDim2.new(0, x, 0, y)
        lines.tl_h.Size = UDim2.new(0, len, 0, thickness)
        lines.tl_v.Position = UDim2.new(0, x, 0, y)
        lines.tl_v.Size = UDim2.new(0, thickness, 0, len)

        lines.tr_h.Position = UDim2.new(0, x + w - len, 0, y)
        lines.tr_h.Size = UDim2.new(0, len, 0, thickness)
        lines.tr_v.Position = UDim2.new(0, x + w - thickness, 0, y)
        lines.tr_v.Size = UDim2.new(0, thickness, 0, len)

        lines.bl_h.Position = UDim2.new(0, x, 0, y + h - thickness)
        lines.bl_h.Size = UDim2.new(0, len, 0, thickness)
        lines.bl_v.Position = UDim2.new(0, x, 0, y + h - len)
        lines.bl_v.Size = UDim2.new(0, thickness, 0, len)

        lines.br_h.Position = UDim2.new(0, x + w - len, 0, y + h - thickness)
        lines.br_h.Size = UDim2.new(0, len, 0, thickness)
        lines.br_v.Position = UDim2.new(0, x + w - thickness, 0, y + h - len)
        lines.br_v.Size = UDim2.new(0, thickness, 0, len)
    end
end

local function getDynamicBoundingBox(character)
    local minX, minY = math.huge, math.huge
    local maxX, maxY = -math.huge, -math.huge
    local anyOnScreen = false

    for _, part in ipairs(character:GetChildren()) do
        if part:IsA("BasePart") then
            local size = part.Size
            local cframe = part.CFrame

            local vertices = {
                cframe * Vector3.new(-size.X/2, -size.Y/2, -size.Z/2),
                cframe * Vector3.new( size.X/2, -size.Y/2, -size.Z/2),
                cframe * Vector3.new(-size.X/2,  size.Y/2, -size.Z/2),
                cframe * Vector3.new(-size.X/2, -size.Y/2,  size.Z/2),
                cframe * Vector3.new( size.X/2,  size.Y/2, -size.Z/2),
                cframe * Vector3.new( size.X/2, -size.Y/2,  size.Z/2),
                cframe * Vector3.new(-size.X/2,  size.Y/2,  size.Z/2),
                cframe * Vector3.new( size.X/2,  size.Y/2,  size.Z/2)
            }

            for i = 1, 8 do
                local screenPos, onScreen = Camera:WorldToViewportPoint(vertices[i])
                if onScreen then
                    anyOnScreen = true
                end
                if screenPos.X < minX then minX = screenPos.X end
                if screenPos.X > maxX then maxX = screenPos.X end
                if screenPos.Y < minY then minY = screenPos.Y end
                if screenPos.Y > maxY then maxY = screenPos.Y end
            end
        end
    end

    if not anyOnScreen then return nil end

    local width = maxX - minX
    local height = maxY - minY

    if width < 5 or height < 5 then return nil end

    return minX, minY, width, height
end

local function processVisuals()
    updateLighting()
    updateAtmosphere()
    updateWeaponSkins()
    updatePlayerSkin()

    if not Config.enabled then
        for player, _ in pairs(espStorage) do
            local data = espStorage[player]
            data.container.Visible = false
            data.nameLabel.Visible = false
            data.healthLabel.Visible = false
            if chamsHighlights[player] then
                chamsHighlights[player]:Destroy()
                chamsHighlights[player] = nil
            end
        end
        return
    end

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            local data = espStorage[player]
            local character = player.Character
            local humanoid = character and character:FindFirstChildOfClass("Humanoid")

            if character and humanoid and data then
                updatePlayerChams(player, character)

                local posX, posY, sizeX, sizeY = getDynamicBoundingBox(character)

                if posX then
                    local padding = 2
                    posX = posX - padding
                    posY = posY - padding
                    sizeX = sizeX + (padding * 2)
                    sizeY = sizeY + (padding * 2)

                    if Config.box then
                        updateBoxPositions(data.lines, posX, posY, sizeX, sizeY, 1.5, Config.box_color)
                        data.container.Visible = true
                        for _, line in pairs(data.lines) do
                            line.BackgroundTransparency = Config.box_transparency
                            local inline = line:FindFirstChildOfClass("Frame")
                            if inline then
                                inline.BackgroundTransparency = math.clamp(Config.box_transparency + 0.2, 0, 1)
                            end
                        end
                    else
                        data.container.Visible = false
                    end

                    if Config.names then
                        data.nameLabel.Text = player.DisplayName or player.Name
                        data.nameLabel.TextColor3 = Config.names_color
                        data.nameLabel.TextTransparency = Config.names_transparency
                        data.nameLabel.TextStrokeTransparency = math.clamp(Config.names_transparency + 0.3, 0, 1)
                        data.nameLabel.Position = UDim2.new(0, posX + (sizeX / 2) - 100, 0, posY - 15)
                        data.nameLabel.TextXAlignment = Enum.TextXAlignment.Center
                        data.nameLabel.Visible = true
                    else
                        data.nameLabel.Visible = false
                    end

                    if Config.health then
                        local hp = math.max(0, math.floor(humanoid.Health))
                        local maxHp = humanoid.MaxHealth > 0 and humanoid.MaxHealth or 100
                        local pct = math.clamp(hp / maxHp, 0, 1)

                        data.healthLabel.Text = hp .. " HP"

                        local chosenColor, chosenTransparency
                        if Config.health_dynamic then
                            chosenColor = Config.health_low_color:Lerp(Config.health_high_color, pct)
                            chosenTransparency = Config.health_low_transparency + (Config.health_high_transparency - Config.health_low_transparency) * pct
                        else
                            chosenColor = Config.health_color
                            chosenTransparency = Config.health_transparency
                        end

                        data.healthLabel.TextColor3 = chosenColor
                        data.healthLabel.TextTransparency = chosenTransparency
                        data.healthLabel.TextStrokeTransparency = math.clamp(chosenTransparency + 0.3, 0, 1)
                        data.healthLabel.Position = UDim2.new(0, posX + sizeX + 5, 0, posY + 1)
                        data.healthLabel.TextXAlignment = Enum.TextXAlignment.Left
                        data.healthLabel.Visible = true
                    else
                        data.healthLabel.Visible = false
                    end
                else
                    data.container.Visible = false
                    data.nameLabel.Visible = false
                    data.healthLabel.Visible = false
                end
            else
                if data then
                    data.container.Visible = false
                    data.nameLabel.Visible = false
                    data.healthLabel.Visible = false
                end
            end
        end
    end
end

local currentH, currentS, currentV, currentAlpha = 0, 0, 0, 0

local applyTheme

local function updatePaletteColor(skipHexBoxUpdate)
    if not activeColorKey then return end
    local selectedColor = Color3.fromHSV(currentH, currentS, currentV)
    Config[activeColorKey] = selectedColor
    Config[activeAlphaKey] = currentAlpha

    alphaBgGradient.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0, selectedColor),
        ColorSequenceKeypoint.new(1, selectedColor)
    })

    if activePreviewRect then
        activePreviewRect.BackgroundColor3 = selectedColor
        activePreviewRect.BackgroundTransparency = currentAlpha
    end

    if not skipHexBoxUpdate then
        hexBox.Text = colorToHex(selectedColor)
    end
    
    if activeColorKey == "menu_color" then
        applyTheme()
    elseif activeColorKey == "lighting_color" then
        updateLighting()
    elseif activeColorKey == "fog_color" then
        updateAtmosphere()
    elseif string.sub(activeColorKey, 1, 11) == "skin_color_" then
        updateWeaponSkins()
    elseif activeColorKey == "player_skin_color" then
        updatePlayerSkin()
    end
end

local function bindExternalColorPicker(colKey, alphaKey, previewRect, contextText)
    if activeColorKey == colKey then
        ColorPickerPanel.Visible = not ColorPickerPanel.Visible
        if not ColorPickerPanel.Visible then
            activeColorKey = nil; activeAlphaKey = nil; activePreviewRect = nil
        end
        return
    end

    activeColorKey = colKey
    activeAlphaKey = alphaKey
    activePreviewRect = previewRect
    CPIndicator.Text = contextText and contextText:upper() or "SELECTING COLOR"
    ColorPickerPanel.Visible = true

    local col = Config[colKey] or Color3.fromRGB(255, 255, 255)
    currentAlpha = Config[alphaKey] or 0
    currentH, currentS, currentV = Color3.toHSV(col)

    colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1)
    baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1))
    hueSlider.Position = UDim2.new(0, -2, currentH, 0)
    pin.Position = UDim2.new(currentS, 0, 1 - currentV, 0)
    alphaSlider.Position = UDim2.new(1 - currentAlpha, -1, 0, -2)
    hexBox.Text = colorToHex(col)

    alphaBgGradient.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0, col),
        ColorSequenceKeypoint.new(1, col)
    })
end

local dragCanvas, dragHue, dragAlpha = false, false, false

canvas.InputBegan:Connect(function(input)
    if not activeColorKey then return end
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragCanvas = true
        local relativeX = math.clamp((input.Position.X - canvas.AbsolutePosition.X) / canvas.AbsoluteSize.X, 0, 1)
        local relativeY = math.clamp((input.Position.Y - canvas.AbsolutePosition.Y) / canvas.AbsoluteSize.Y, 0, 1)
        currentS = relativeX
        currentV = 1 - relativeY
        pin.Position = UDim2.new(relativeX, 0, relativeY, 0)
        updatePaletteColor(false)
    end
end)

hueStrip.InputBegan:Connect(function(input)
    if not activeColorKey then return end
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragHue = true
        local relativeY = math.clamp((input.Position.Y - hueStrip.AbsolutePosition.Y) / hueStrip.AbsoluteSize.Y, 0, 1)
        currentH = relativeY
        hueSlider.Position = UDim2.new(0, -2, relativeY, 0)
        colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1)
        baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1))
        updatePaletteColor(false)
    end
end)

alphaStrip.InputBegan:Connect(function(input)
    if not activeColorKey then return end
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragAlpha = true
        local relativeX = math.clamp((input.Position.X - alphaStrip.AbsolutePosition.X) / alphaStrip.AbsoluteSize.X, 0, 1)
        currentAlpha = 1 - relativeX
        alphaSlider.Position = UDim2.new(relativeX, -1, 0, -2)
        updatePaletteColor(false)
    end
end)

hexBox.FocusLost:Connect(function()
    if not activeColorKey then return end
    local targetColor = hexToColor(hexBox.Text)
    if targetColor then
        currentH, currentS, currentV = Color3.toHSV(targetColor)
        colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1)
        baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1))
        hueSlider.Position = UDim2.new(0, -2, currentH, 0)
        pin.Position = UDim2.new(currentS, 0, 1 - currentV, 0)
        updatePaletteColor(true)
        hexBox.Text = colorToHex(targetColor)
    else
        hexBox.Text = colorToHex(Config[activeColorKey])
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragCanvas = false; dragHue = false; dragAlpha = false
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        if dragCanvas and ColorPickerPanel.Visible then
            local relativeX = math.clamp((input.Position.X - canvas.AbsolutePosition.X) / canvas.AbsoluteSize.X, 0, 1)
            local relativeY = math.clamp((input.Position.Y - canvas.AbsolutePosition.Y) / canvas.AbsoluteSize.Y, 0, 1)
            currentS = relativeX
            currentV = 1 - relativeY
            pin.Position = UDim2.new(relativeX, 0, relativeY, 0)
            updatePaletteColor(false)
        elseif dragHue and ColorPickerPanel.Visible then
            local relativeY = math.clamp((input.Position.Y - hueStrip.AbsolutePosition.Y) / hueStrip.AbsoluteSize.Y, 0, 1)
            currentH = relativeY
            hueSlider.Position = UDim2.new(0, -2, relativeY, 0)
            colorLayer.BackgroundColor3 = Color3.fromHSV(currentH, 1, 1)
            baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH, 1, 1))
            updatePaletteColor(false)
        elseif dragAlpha and ColorPickerPanel.Visible then
            local relativeX = math.clamp((input.Position.X - alphaStrip.AbsolutePosition.X) / alphaStrip.AbsoluteSize.X, 0, 1)
            currentAlpha = 1 - relativeX
            alphaSlider.Position = UDim2.new(relativeX, -1, 0, -2)
            updatePaletteColor(false)
        end
    end
end)

local function addVisualRow(labelText, configKey, parentFrame, colorMode, colorData)

    row.BackgroundColor3 = Color3.fromRGB(32, 32, 32)

    checkbox.BackgroundColor3 = Config[configKey] and Config.menu_color or Color3.fromRGB(42, 42, 42)
    checkbox.BorderSizePixel = 0
    checkbox.Text = ""
    checkbox.Parent = row

    local checkboxStroke = Instance.new("UIStroke")
    checkboxStroke.Thickness = 1
    checkboxStroke.Color = Config[configKey] and Config.menu_color or Color3.fromRGB(62, 62, 62)
    checkboxStroke.Parent = checkbox

    local checkMark = Instance.new("TextLabel")
    checkMark.Size = UDim2.new(1, 0, 1, -1)
    checkMark.BackgroundTransparency = 1
    checkMark.Text = "✓"
    checkMark.TextColor3 = Color3.fromRGB(255, 255, 255)
    checkMark.Font = Enum.Font.GothamBold
    checkMark.TextSize = 10
    checkMark.Visible = Config[configKey]
    checkMark.Parent = checkbox

    local lbl = Instance.new("TextLabel")
    lbl.Size = UDim2.new(0, 180, 1, 0)
    lbl.Position = UDim2.new(0, 36, 0, 0)
    lbl.BackgroundTransparency = 1
    lbl.Text = labelText
    lbl.TextColor3 = Config[configKey] and Color3.fromRGB(240, 240, 245) or Color3.fromRGB(140, 140, 150)
    lbl.TextXAlignment = Enum.TextXAlignment.Left
    lbl.Font = Enum.Font.Gotham
    lbl.TextSize = 12
    lbl.Parent = row

    table.insert(activeCheckboxes, {box = checkbox, stroke = checkboxStroke, key = configKey, label = lbl})

    local function setupColorButton(colKey, alphaKey, posXOffset, contextName)
        local colorPreview = Instance.new("TextButton")
        colorPreview.Size = UDim2.new(0, 18, 0, 12)
        colorPreview.Position = UDim2.new(1, posXOffset, 0.5, -6)
        colorPreview.BackgroundColor3 = Config[colKey] or Color3.fromRGB(255, 255, 255)
        colorPreview.BackgroundTransparency = Config[alphaKey] or 0
        colorPreview.BorderSizePixel = 0
        colorPreview.Text = ""
        colorPreview.Parent = row

        local previewStroke = Instance.new("UIStroke")
        previewStroke.Thickness = 1
        previewStroke.Color = Color3.fromRGB(45, 45, 55)
        previewStroke.Parent = colorPreview

        colorPreview.MouseButton1Click:Connect(function()
            bindExternalColorPicker(colKey, alphaKey, colorPreview, contextName)
        end)
    end

    if colorMode == "single" and colorData then
        setupColorButton(colorData.colKey, colorData.alphaKey, -30, colorData.context)
    elseif colorMode == "dual" and colorData then
        setupColorButton(colorData[1].colKey, colorData[1].alphaKey, -54, colorData[1].context)
        setupColorButton(colorData[2].colKey, colorData[2].alphaKey, -30, colorData[2].context)
    end

    checkbox.MouseButton1Click:Connect(function()
        Config[configKey] = not Config[configKey]
        checkMark.Visible = Config[configKey]
        checkbox.BackgroundColor3 = Config[configKey] and Config.menu_color or Color3.fromRGB(42, 42, 42)
        checkboxStroke.Color = Config[configKey] and Config.menu_color or Color3.fromRGB(62, 62, 62)
        lbl.TextColor3 = Config[configKey] and Color3.fromRGB(240, 240, 245) or Color3.fromRGB(140, 140, 150)
    end)

    return row
end

    local relativeX = math.clamp((input.Position.X - LightingBrightnessSliderContainer.AbsolutePosition.X) / LightingBrightnessSliderContainer.AbsoluteSize.X, 0, 1)
    local brightness = math.round((relativeX * 5) * 10) / 10
    Config.lighting_brightness = brightness
    LightingBrightnessSliderFill.Size = UDim2.new(relativeX, 0, 1, 0)
    LightingBrightnessValue.Text = tostring(brightness)
    updateLighting()
end

    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        draggingLightingBrightness = true
        updateLightingBrightnessSlider(input)
    end
end)

    if draggingLightingBrightness and input.UserInputType == Enum.UserInputType.MouseMovement then
        updateLightingBrightnessSlider(input)
    end

    if input.UserInputType == Enum.UserInputType.MouseButton1 then

-- ==================== LIVE THEME REFRESH LOGIC ====================

applyTheme = function()
    local clr = Config.menu_color
    if AccentLine then AccentLine.BackgroundColor3 = clr end
    if CPAccent then CPAccent.BackgroundColor3 = clr end
    if hexBox then hexBox.TextColor3 = clr end
    if CPIndicator then CPIndicator.TextColor3 = clr end
    if UIRefs.MenuColorPreview then
        UIRefs.MenuColorPreview.BackgroundColor3 = clr
        UIRefs.MenuColorPreview.BackgroundTransparency = Config.menu_transparency
    end
    if WatermarkTitle then WatermarkTitle.TextColor3 = clr end
    if WatermarkStats then WatermarkStats.TextColor3 = clr end
    
    if VisualsInd then VisualsInd.BackgroundColor3 = clr end
    if SettingsInd then SettingsInd.BackgroundColor3 = clr end
    if VisualsTabBtn then
        VisualsTabBtn.BackgroundColor3 = VisualsPage.Visible and clr or Color3.fromRGB(0, 0, 0)
        VisualsTabBtn.TextColor3 = VisualsPage.Visible and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(145, 145, 145)
        VisualsTabBtn.BackgroundTransparency = VisualsPage.Visible and 0 or 1
    end
    if SettingsTabBtn then
        SettingsTabBtn.BackgroundColor3 = SettingsPage.Visible and clr or Color3.fromRGB(0, 0, 0)
        SettingsTabBtn.TextColor3 = SettingsPage.Visible and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(145, 145, 145)
        SettingsTabBtn.BackgroundTransparency = SettingsPage.Visible and 0 or 1
    end
    
    for _, data in pairs(subTabs) do
        local active = data.page.Visible
        data.btn.TextColor3 = active and clr or Color3.fromRGB(140, 140, 150)
        data.stroke.Color = active and clr or Color3.fromRGB(44, 44, 44)
    end
    
    for _, header in ipairs(collapsibleHeaders) do
        if header:IsA("TextButton") then
            header.TextColor3 = clr
        elseif header:IsA("TextLabel") then
            header.TextColor3 = clr
        end
    end
    
    for _, cb in ipairs(activeCheckboxes) do
        if Config[cb.key] then
            cb.box.BackgroundColor3 = clr
            cb.stroke.Color = clr
            cb.label.TextColor3 = Color3.fromRGB(240, 240, 240)
        else
            cb.box.BackgroundColor3 = Color3.fromRGB(42, 42, 42)
            cb.stroke.Color = Color3.fromRGB(62, 62, 62)
            cb.label.TextColor3 = Color3.fromRGB(145, 145, 145)
        end
    end
    
    if UIRefs.BindButton and not UIRefs.listeningForBind then
        UIRefs.BindButton.TextColor3 = clr
    end
    
    if BoxStyleBtn then
        BoxStyleBtn.TextColor3 = clr
    end

    if UIRefs.SliderFill then
        UIRefs.SliderFill.BackgroundColor3 = clr
    end
    if UIRefs.LightingBrightnessSliderFill then
        UIRefs.LightingBrightnessSliderFill.BackgroundColor3 = clr
    end
    if UIRefs.WeaponSkinWeaponButton then
        UIRefs.WeaponSkinWeaponButton.TextColor3 = clr
    end
    if UIRefs.WeaponSkinMaterialButton then
        UIRefs.WeaponSkinMaterialButton.TextColor3 = clr
    end
    if UIRefs.WeaponSkinColorPreview then
        UIRefs.WeaponSkinColorPreview.BackgroundColor3 = Config["skin_color_" .. WeaponSkinIds[WeaponSkinSelected]]
        UIRefs.WeaponSkinColorPreview.BackgroundTransparency = Config["skin_transparency_" .. WeaponSkinIds[WeaponSkinSelected]]
    end
end

-- ==================== SYSTEM TRANSITION ANIMATION ====================

local transitionActive = false

local function toggleMenu()
    if transitionActive then return end
    transitionActive = true
    menuVisible = not menuVisible

    if menuVisible then
        MainFrame.Visible = true
        TweenService:Create(MainFrame, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        TweenService:Create(Sidebar, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        TweenService:Create(Logo, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {TextTransparency = 0}):Play()
        TweenService:Create(MainStroke, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0}):Play()
        TweenService:Create(SepLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        TweenService:Create(AccentLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        task.wait(0.15)
    else
        TweenService:Create(MainFrame, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        TweenService:Create(Sidebar, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        TweenService:Create(Logo, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {TextTransparency = 1}):Play()
        TweenService:Create(MainStroke, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {Transparency = 1}):Play()
        TweenService:Create(SepLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        TweenService:Create(AccentLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        task.wait(0.15)
        if not menuVisible then
            MainFrame.Visible = false
            ColorPickerPanel.Visible = false
            activeColorKey = nil; activeAlphaKey = nil; activePreviewRect = nil
        end
    end
    transitionActive = false
end

    if player ~= LocalPlayer then createDrawings(player) end
end))

    removeDrawings(player)
end))

end

    processVisuals()
    updateWatermark(dt)
end))

table.insert(connections, UserInputService.InputBegan:Connect(function(input, processed)
    if UIRefs.listeningForBind and input.UserInputType == Enum.UserInputType.Keyboard then
        Config.toggleKey = input.KeyCode
        UIRefs.BindButton.Text = input.KeyCode.Name
        UIRefs.BindButton.TextColor3 = Config.menu_color
        UIRefs.listeningForBind = false
        return
    end

    if processed then return end
    if input.KeyCode == Config.toggleKey then
        toggleMenu()
    end
end))

            if data.healthLabel then data.healthLabel:Destroy() end
        end
    end
end)
