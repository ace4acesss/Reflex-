function updateArrayList()
    if not Config.array_list then
        if ArrayListFrame.Visible then
            ArrayListFrame.Visible = false
            for k, v in pairs(ArrayListCacheLabels) do v:Destroy() end
            table.clear(ArrayListCacheLabels)
        end
        return
    end
    ArrayListFrame.BackgroundTransparency = Config.array_list_box and 0 or 1
    ArrayListStroke.Enabled = Config.array_list_box

    local idx = 1
    for _, mod in ipairs(ArrayListModules) do
        if Config[mod.configKey] then
            local lbl = ArrayListCacheLabels[mod.configKey]
            if not lbl then
                lbl = Instance.new("TextLabel")
                lbl.BackgroundTransparency = 1
                lbl.AutomaticSize = Enum.AutomaticSize.XY
                lbl.Font = Enum.Font.GothamBold
                lbl.TextSize = 12
                lbl.TextXAlignment = Enum.TextXAlignment.Right
                lbl.Text = mod.label
                lbl.Parent = ArrayListFrame
                local st = Instance.new("UIStroke")
                st.Thickness = 1.25
                st.Color = Color3.fromRGB(30,30,30)
                st.Parent = lbl
                applyTextShine(lbl)
                ArrayListCacheLabels[mod.configKey] = lbl
            end
            local col = Config.menu_color
            if mod.bindID then
                local pressed = (mod.bindID == "aimbot" and aimbotKeyPressed) or (mod.bindID == "fakelag" and fakelagKeyPressed)
                if not pressed then col = Color3.fromRGB(140,140,150) end
            end
            lbl.TextColor3 = col
            lbl.LayoutOrder = idx
            idx = idx + 1
        else
            local lbl = ArrayListCacheLabels[mod.configKey]
            if lbl then lbl:Destroy(); ArrayListCacheLabels[mod.configKey] = nil end
        end
    end
    ArrayListFrame.Visible = (idx > 1)
end

fpsValue = 0; fpsAccumulator = 0; fpsFrames = 0; hudAccumulator = 0

function updateHUDOverlay(dt)
    fpsAccumulator = fpsAccumulator + dt
    fpsFrames = fpsFrames + 1
    hudAccumulator = hudAccumulator + dt
    if fpsAccumulator >= 0.5 then
        fpsValue = math.floor(fpsFrames / fpsAccumulator + 0.5)
        fpsAccumulator = 0; fpsFrames = 0
    end
    if hudAccumulator >= 0.2 then
        hudAccumulator = 0
        if Config.watermark then
            WatermarkFrame.BackgroundTransparency = Config.watermark_box and 0 or 1
            WatermarkStroke.Enabled = Config.watermark_box
            WatermarkFrame.Visible = true
            WatermarkTitle.TextColor3 = Config.menu_color
            WatermarkStats.TextColor3 = Config.menu_color
            WatermarkAvatar.Visible = Config.watermark_pfp
            local parts = {}
            if Config.watermark_fps then table.insert(parts, tostring(fpsValue) .. " FPS") end
            if Config.watermark_ping then
                pcall(function() table.insert(parts, math.floor(LocalPlayer:GetNetworkPing() * 1000 + 0.5) .. " MS") end)
            end
            if Config.watermark_name then table.insert(parts, LocalPlayer.DisplayName or LocalPlayer.Name) end
            WatermarkStats.Text = table.concat(parts, "  |  ")
            WatermarkStats.Visible = #parts > 0
            if #parts == 0 and not Config.watermark_pfp then
                WatermarkTitle.Size = UDim2.new(0, 76, 0, 18)
                WatermarkTitle.TextXAlignment = Enum.TextXAlignment.Center
            else
                WatermarkTitle.Size = UDim2.new(0, 64, 0, 18)
                WatermarkTitle.TextXAlignment = Enum.TextXAlignment.Left
            end
        else
            WatermarkFrame.Visible = false
        end
    end
end

function colorToHex(c)
    local r = math.round(c.R * 255); local g = math.round(c.G * 255); local b = math.round(c.B * 255)
    return string.format("#%02X%02X%02X", r, g, b)
end

function hexToColor(h)
    local ok, c = pcall(function() return Color3.fromHex(h) end)
    return ok and c or nil
end

function matchesBind(input, bind)
    if typeof(bind) == "EnumItem" then
        if bind.EnumType == Enum.KeyCode then return input.KeyCode == bind
        elseif bind.EnumType == Enum.UserInputType then return input.UserInputType == bind end
    end
    return false
end

function getBindName(bind) return typeof(bind) == "EnumItem" and bind.Name or "None" end

function GetCorrectAnimID(char, packString, defR15, defR6)
    local ani = char:FindFirstChild("Animate") or char:FindFirstChild("animate")
    if ani then
        local f = ani:FindFirstChild(packString)
        if f then
            local a = f:FindFirstChildWhichIsA("Animation")
            if a and a.AnimationId ~= "" then return a.AnimationId end
        end
    end
    local hum = char:FindFirstChild("Humanoid")
    if hum and hum.RigType == Enum.HumanoidRigType.R6 then return defR6 end
    return defR15
end

function PlaySyncAnims(name)
    for _, arr in pairs({FakeAnims, RealAnims}) do
        if arr[name] and not arr[name].IsPlaying then arr[name]:Play(0.1) end
    end
end

function StopSyncAnims()
    for _, arr in pairs({FakeAnims, RealAnims}) do
        for _, t in pairs(arr) do if t and t.IsPlaying then t:Stop(0.1) end end
    end
end

function ShutdownGhostSystemGracefully()
    for _, c in pairs(FakelagHooks) do c:Disconnect() end
    table.clear(FakelagHooks)
    StopSyncAnims()
    if FakeChar then
        local returnCF = FakeChar:GetPrimaryPartCFrame()
        FakeChar:Destroy(); FakeChar = nil
        if RealChar then
            local rHum = RealChar:FindFirstChild("Humanoid")
            if rHum then rHum.WalkSpeed = 16; rHum.JumpPower = 50; rHum.PlatformStand = false end
            RealChar:PivotTo(returnCF)
            local rRoot = RealChar:FindFirstChild("HumanoidRootPart")
            if rRoot then rRoot.Velocity = Vector3.zero; rRoot.Anchored = false end
            for item, orig in pairs(RealPartOriginalStates) do
                if item and item.Parent then item.Transparency = orig end
            end
            local animScript = RealChar:FindFirstChild("Animate") or RealChar:FindFirstChild("animate")
            if animScript and animScript:IsA("LocalScript") then animScript.Enabled = true end
            table.clear(RealPartOriginalStates); table.clear(RealAnims); table.clear(FakeAnims)
            Camera.CameraSubject = RealChar:FindFirstChild("Humanoid")
        end
    end
end

function ActivateFakelagGhost()
    RealChar = LocalPlayer.Character
    if not RealChar or not RealChar.PrimaryPart or not RealChar:FindFirstChild("HumanoidRootPart") then return false end
    CurrentGhostPosServerCacheCF = RealChar.PrimaryPart.CFrame
    RealChar.Archivable = true
    FakeChar = RealChar:Clone()
    FakeChar:PivotTo(CurrentGhostPosServerCacheCF)
    FakeChar.Name = "LocalOverrideCloneTrackerClientGhost"

    local realHum = RealChar:FindFirstChild("Humanoid")
    local realRoot = RealChar:FindFirstChild("HumanoidRootPart")
    if realHum then realHum.WalkSpeed = 0; realHum.JumpPower = 0 end
    if realRoot then realRoot.Anchored = false end

    local origAnim = RealChar:FindFirstChild("Animate") or RealChar:FindFirstChild("animate")
    if origAnim and origAnim:IsA("LocalScript") then origAnim.Enabled = false end

    local solidParts = {}
    table.clear(RealPartOriginalStates)
    for _, d in ipairs(RealChar:GetDescendants()) do
        if d:IsA("BasePart") or d:IsA("Decal") or d:IsA("Texture") then
            RealPartOriginalStates[d] = d.Transparency
            if d.Name == "HumanoidRootPart" then d.Transparency = 1 else d.Transparency = 0.5 end
            if d:IsA("BasePart") then table.insert(solidParts, d) end
        end
    end
    for _, f in ipairs(FakeChar:GetDescendants()) do
        if f:IsA("BaseScript") then f:Destroy() end
        if f:IsA("BasePart") then
            f.Anchored = false
            f.CanCollide = (f.Name == "HumanoidRootPart")
            if f.Name == "HumanoidRootPart" then f.Transparency = 1 end
            for _, p in ipairs(solidParts) do
                local nc = Instance.new("NoCollisionConstraint")
                nc.Part0 = p; nc.Part1 = f; nc.Parent = f
            end
        end
    end

    FakeChar.Parent = workspace
    local fakeHum = FakeChar:FindFirstChild("Humanoid")

    local idleSrc = Instance.new("Animation"); idleSrc.AnimationId = GetCorrectAnimID(RealChar, "idle", "rbxassetid://507766388", "rbxassetid://180435571")
    local walkSrc = Instance.new("Animation"); walkSrc.AnimationId = GetCorrectAnimID(RealChar, "walk", "rbxassetid://507777826", "rbxassetid://180426354")
    local runSrc  = Instance.new("Animation"); runSrc.AnimationId  = GetCorrectAnimID(RealChar, "run",  "rbxassetid://507767714", "rbxassetid://180426354")
    local jpSrc   = Instance.new("Animation"); jpSrc.AnimationId   = GetCorrectAnimID(RealChar, "jump", "rbxassetid://507765000", "rbxassetid://125750702")
    local fallSrc = Instance.new("Animation"); fallSrc.AnimationId = GetCorrectAnimID(RealChar, "freefall", "rbxassetid://507767968", "rbxassetid://180436148")
    if runSrc.AnimationId == "" then runSrc.AnimationId = walkSrc.AnimationId end

    local fakeAnim = fakeHum:FindFirstChildOfClass("Animator") or Instance.new("Animator", fakeHum)
    FakeAnims.Idle = fakeAnim:LoadAnimation(idleSrc); FakeAnims.Run = fakeAnim:LoadAnimation(runSrc)
    FakeAnims.Fall = fakeAnim:LoadAnimation(fallSrc); FakeAnims.Jump = fakeAnim:LoadAnimation(jpSrc)

    local realAnim = realHum:FindFirstChildOfClass("Animator") or Instance.new("Animator", realHum)
    RealAnims.Idle = realAnim:LoadAnimation(idleSrc); RealAnims.Run = realAnim:LoadAnimation(runSrc)
    RealAnims.Fall = realAnim:LoadAnimation(fallSrc); RealAnims.Jump = realAnim:LoadAnimation(jpSrc)

    Camera.CameraSubject = fakeHum
    PlaySyncAnims("Idle")

    table.insert(FakelagHooks, fakeHum.StateChanged:Connect(function(_, newState)
        if newState == Enum.HumanoidStateType.Jumping then StopSyncAnims(); PlaySyncAnims("Jump")
        elseif newState == Enum.HumanoidStateType.Freefall then StopSyncAnims(); PlaySyncAnims("Fall")
        elseif newState == Enum.HumanoidStateType.Landed then StopSyncAnims(); PlaySyncAnims("Idle") end
    end))
    table.insert(FakelagHooks, RunService.Stepped:Connect(function()
        if fakelagSystemActiveNow and RealChar and RealChar.PrimaryPart then
            RealChar.PrimaryPart.CFrame = CurrentGhostPosServerCacheCF
            RealChar.PrimaryPart.AssemblyLinearVelocity = Vector3.zero
            RealChar.PrimaryPart.AssemblyAngularVelocity = Vector3.zero
        end
    end))
    table.insert(FakelagHooks, UserInputService.JumpRequest:Connect(function()
        if fakelagSystemActiveNow and fakeHum and fakeHum:GetState() ~= Enum.HumanoidStateType.Freefall then fakeHum.Jump = true end
    end))

    fakelagAccumulator = 0
    return true
end

function processFakelagEngineRenderCycle(dt)
    if Config.fakelag and fakelagKeyPressed and not fakelagSystemActiveNow then
        if ActivateFakelagGhost() then fakelagSystemActiveNow = true end
    elseif (not Config.fakelag or not fakelagKeyPressed) and fakelagSystemActiveNow then
        ShutdownGhostSystemGracefully()
        fakelagSystemActiveNow = false
    end

    if fakelagSystemActiveNow and FakeChar then
        fakelagAccumulator = fakelagAccumulator + dt
        if fakelagAccumulator >= Config.fakelag_delay then
            fakelagAccumulator = 0
            if FakeChar.PrimaryPart then CurrentGhostPosServerCacheCF = FakeChar.PrimaryPart.CFrame end
        end
        local fakeHum = FakeChar:FindFirstChild("Humanoid")
        if fakeHum then
            fakeHum.WalkSpeed = Config.fakelag_speed
            local inputVector = Vector3.zero
            if Controls then
                inputVector = Controls:GetMoveVector()
            else
                if UserInputService:IsKeyDown(Enum.KeyCode.W) then inputVector += Vector3.new(0,0,-1) end
                if UserInputService:IsKeyDown(Enum.KeyCode.S) then inputVector += Vector3.new(0,0,1) end
                if UserInputService:IsKeyDown(Enum.KeyCode.A) then inputVector += Vector3.new(-1,0,0) end
                if UserInputService:IsKeyDown(Enum.KeyCode.D) then inputVector += Vector3.new(1,0,0) end
            end
            local target = (Camera.CFrame.RightVector * inputVector.X) + (Camera.CFrame.LookVector * -inputVector.Z)
            target = Vector3.new(target.X, 0, target.Z)
            local state = fakeHum:GetState()
            local grounded = (state == Enum.HumanoidStateType.Running or state == Enum.HumanoidStateType.RunningNoPhysics)
            if target.Magnitude > 0 then
                fakeHum:Move(target.Unit, false)

                fakeHum:Move(Vector3.zero, false)
                if grounded and FakeAnims.Idle and not FakeAnims.Idle.IsPlaying then StopSyncAnims(); PlaySyncAnims("Idle") end
            end
        end
    end
end

-- ==================== TEAM CHECK ====================
function isLocalHoldingTaser()
    local char = LocalPlayer.Character
    if not char then return false end
    for _, t in ipairs(char:GetChildren()) do
        if t:IsA("Tool") and (t:GetAttribute("Behavior") == "Taser" or t:GetAttribute("Projectile") == "Taser" or string.find(string.lower(t.Name), "taser")) then
            return true
        end
    end
    return false
end

function passesTeamCheck(target)
    if not Config.team_check then return true end
    local tm = target.Team
    if not tm then return true end
    local nm = string.lower(tm.Name)
    local isGuard = string.find(nm, "guard") or string.find(nm, "cop") or string.find(nm, "police")
    local isInmate = string.find(nm, "inmate") or string.find(nm, "prisoner")
    local isCrim = string.find(nm, "criminal")
    if Config.ignore_guards and isGuard then return false end
    if Config.ignore_criminals and isCrim then return false end
    if Config.ignore_inmates and isInmate then
        local ch = target.Character
        if ch then
            if Config.target_hostile and ch:GetAttribute("Hostile") then return true end
            if Config.target_trespassing and ch:GetAttribute("Trespassing") then return true end
        end
        return false
    end
    return true
end

-- ==================== AIMBOT / SILENT AIM ====================
function checkVisibility(player, character)
    if not Config.wall_check then return true end
    local root = character:FindFirstChild("HumanoidRootPart")
    if not root then return false end
    local camPos = Camera.CFrame.Position
    local dir = root.Position - camPos
    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    local filter = {character}
    if LocalPlayer.Character then table.insert(filter, LocalPlayer.Character) end
    if FakeChar then table.insert(filter, FakeChar) end
    params.FilterDescendantsInstances = filter
    local result = workspace:Raycast(camPos, dir, params)
    return not result
end

function getClosestTarget()
    local closest = nil; local shortest = math.huge
    local mousePos = UserInputService:GetMouseLocation()
    for _, pl in ipairs(Players:GetPlayers()) do
        if pl ~= LocalPlayer and passesTeamCheck(pl) then
            local ch = pl.Character
            local hum = ch and ch:FindFirstChildOfClass("Humanoid")
            if ch and hum and hum.Health > 0 then
                local part = ch:FindFirstChild(Config.aimbot_target)
                if part then
                    local pos, on = Camera:WorldToViewportPoint(part.Position)
                    if on then
                        local dist = (Vector2.new(pos.X, pos.Y) - mousePos).Magnitude
                        if dist < Config.aimbot_radius and dist < shortest and checkVisibility(pl, ch) then
                            shortest = dist; closest = pl
                        end
                    end
                end
            end
        end
    end
    return closest
end

function aimAtTarget(target)
    if not target or not target.Character then return end
    local part = target.Character:FindFirstChild(Config.aimbot_target)
    if part then Camera.CFrame = CFrame.new(Camera.CFrame.Position, part.Position) end
end

-- CastRay hook (silent aim)
function noUpvals(fn) return function(...) return fn(...) end end

origCastRay = nil
hooked = false

function setupHook()
    local castRayFunc = nil
    if type(filtergc) == "function" then castRayFunc = filtergc("function", {Name = "castRay"}, true)
    elseif type(getgc) == "function" then
        for _, v in pairs(getgc(true)) do
            if type(v) == "function" and debug.getinfo(v).name == "castRay" then castRayFunc = v; break end
        end
    end
    if not castRayFunc then return false end
    origCastRay = hookfunction(castRayFunc, noUpvals(function(startPos, targetPos, ...)
        if not checkcaller() and aimbotKeyPressed and Config.silent_aim and silentAimTarget and silentAimTarget.Character then
            local tgtPart = silentAimTarget.Character:FindFirstChild(Config.aimbot_target) or silentAimTarget.Character:FindFirstChild("Head")
            if tgtPart then return tgtPart, tgtPart.Position end
        end
        return origCastRay(startPos, targetPos, ...)
    end))
    return true
end

task.spawn(function()
    while not hooked do
        if setupHook() then hooked = true end
        task.wait(0.5)
    end
end)

WeaponSkinSelected = "AK47"

    WeaponSkinIds[w] = id
    Config["skin_enabled_" .. id] = false

    Config["skin_transparency_" .. id] = 0
    Config["skin_material_" .. id] = "Plastic"

function getMaterialByName(name)
    local ok, m = pcall(function() return Enum.Material[tostring(name)] end)
    if ok and m then return m end
    for _, mat in ipairs(Enum.Material:GetEnumItems()) do
        if string.lower(mat.Name) == string.lower(tostring(name)) then return mat end
    end
    return Enum.Material.Plastic
end

function getWeaponRoots()
    local r = {}
    local bp = LocalPlayer:FindFirstChildOfClass("Backpack")
    if bp then table.insert(r, bp) end
    if LocalPlayer.Character then table.insert(r, LocalPlayer.Character) end
    return r
end

function namesMatch(a,b) return string.lower(a) == string.lower(b) end

function findWeaponInstances(wname)
    local found = {}
    for _, root in ipairs(getWeaponRoots()) do
        for _, item in ipairs(root:GetDescendants()) do
            if namesMatch(item.Name, wname) and (item:IsA("Tool") or item:IsA("Model")) then
                table.insert(found, item)
            end
        end
    end
    return found
end

function captureOriginal(wname, mesh, paint)
    local orig = WeaponSkinOriginals[wname]
    if orig[mesh] then return end
    local prop = mesh:IsA("MeshPart") and "TextureID" or "TextureId"
    local tex = ""
    pcall(function() tex = mesh[prop] end)
    orig[mesh] = { textureProperty = prop, texture = tex, part = paint, color = paint and paint.Color or nil, material = paint and paint.Material or nil }
end

function applySkinToMesh(wname, mesh)
    local id = WeaponSkinIds[wname]
    local paint = mesh:IsA("BasePart") and mesh or mesh.Parent
    if not paint or not paint:IsA("BasePart") then return end
    captureOriginal(wname, mesh, paint)
    pcall(function()
        if mesh:IsA("MeshPart") then mesh.TextureID = "" else mesh.TextureId = "" end
    end)
    paint.Color = Config["skin_color_" .. id]
    paint.Material = getMaterialByName(Config["skin_material_" .. id])
end

function restoreWeaponSkin(wname)
    local orig = WeaponSkinOriginals[wname]
    if not orig then return end
    for mesh, data in pairs(orig) do
        if mesh and mesh.Parent then
            pcall(function() mesh[data.textureProperty] = data.texture or "" end)
            if data.part and data.part.Parent then
                if data.color then data.part.Color = data.color end
                if data.material then data.part.Material = data.material end
            end
        end
    end
    table.clear(orig)
end

function restoreAllWeaponSkins()
    for _, w in ipairs(WeaponSkinNames) do restoreWeaponSkin(w) end
end

function applyWeaponSkin(wname)
    local id = WeaponSkinIds[wname]
    if not Config.custom_weapon_skin or not Config["skin_enabled_" .. id] then
        restoreWeaponSkin(wname); return
    end
    for _, weapon in ipairs(findWeaponInstances(wname)) do
        for _, item in ipairs(weapon:GetDescendants()) do
            if item:IsA("MeshPart") or item:IsA("SpecialMesh") then applySkinToMesh(wname, item) end
        end
    end
end

function updateWeaponSkins()
    if not Config.custom_weapon_skin then restoreAllWeaponSkins(); return end
    for _, w in ipairs(WeaponSkinNames) do applyWeaponSkin(w) end
end

-- Player Skin
PlayerSkinOriginals = { clothes = {}, parts = {}, meshes = {}, decals = {} }

function clearPlayerSkinCache()
    for k in pairs(PlayerSkinOriginals.clothes) do PlayerSkinOriginals.clothes[k] = nil end
    for k in pairs(PlayerSkinOriginals.parts) do PlayerSkinOriginals.parts[k] = nil end
    for k in pairs(PlayerSkinOriginals.meshes) do PlayerSkinOriginals.meshes[k] = nil end
    for k in pairs(PlayerSkinOriginals.decals) do PlayerSkinOriginals.decals[k] = nil end
end

function restorePlayerSkin()
    for cloth, parent in pairs(PlayerSkinOriginals.clothes) do
        if cloth and not cloth.Parent and parent and parent.Parent then cloth.Parent = parent end
    end
    for part, data in pairs(PlayerSkinOriginals.parts) do
        if part and part.Parent then
            part.Color = data.color; part.Material = data.material
        end
        local decals = PlayerSkinOriginals.decals[part]
        if decals then
            for _, info in ipairs(decals) do
                if info.obj and info.obj.Parent then
                    info.obj.Transparency = info.transparency
                    pcall(function()
                        if info.obj:IsA("Decal") or info.obj:IsA("Texture") then info.obj.Texture = info.texture
                        elseif info.obj:IsA("SurfaceAppearance") then
                            info.obj.ColorMap = info.colorMap; info.obj.NormalMap = info.normalMap
                            info.obj.MetalnessMap = info.metalnessMap; info.obj.RoughnessMap = info.roughnessMap
                        end
                    end)
                end
            end
        end
    end
    for mesh, data in pairs(PlayerSkinOriginals.meshes) do
        if mesh and mesh.Parent then
            pcall(function() mesh[data.textureProperty] = data.texture or "" end)
        end
    end
    clearPlayerSkinCache()
end

function capturePlayerPart(part)
    if PlayerSkinOriginals.parts[part] then return end
    PlayerSkinOriginals.parts[part] = { color = part.Color, material = part.Material }
    local decals = {}
    for _, child in ipairs(part:GetChildren()) do
        if child:IsA("Decal") or child:IsA("Texture") or child:IsA("SurfaceAppearance") then
            local info = { obj = child, transparency = child.Transparency }
            if child:IsA("Decal") or child:IsA("Texture") then info.texture = child.Texture
            elseif child:IsA("SurfaceAppearance") then
                info.colorMap = child.ColorMap; info.normalMap = child.NormalMap
                info.metalnessMap = child.MetalnessMap; info.roughnessMap = child.RoughnessMap
            end
            table.insert(decals, info)
            child.Transparency = 1
            pcall(function()
                if child:IsA("Decal") or child:IsA("Texture") then child.Texture = ""
                elseif child:IsA("SurfaceAppearance") then
                    child.ColorMap = ""; child.NormalMap = ""; child.MetalnessMap = ""; child.RoughnessMap = ""
                end
            end)
        end
    end
    if next(decals) then PlayerSkinOriginals.decals[part] = decals end
end

function capturePlayerMesh(mesh)
    if PlayerSkinOriginals.meshes[mesh] then return end
    local prop = mesh:IsA("MeshPart") and "TextureID" or "TextureId"
    local tex = ""
    pcall(function() tex = mesh[prop] end)
    PlayerSkinOriginals.meshes[mesh] = { textureProperty = prop, texture = tex }
end

function applyPlayerSkin()
    local char = LocalPlayer.Character
    if not char then return end
    for _, item in ipairs(char:GetChildren()) do
        if item:IsA("Shirt") or item:IsA("Pants") or item:IsA("ShirtGraphic") then
            if not PlayerSkinOriginals.clothes[item] then PlayerSkinOriginals.clothes[item] = char end
            item.Parent = nil
        end
    end
    local mat = getMaterialByName(Config.player_skin_material)
    for _, item in ipairs(char:GetDescendants()) do
        if not item:FindFirstAncestorWhichIsA("Tool") then
            if item:IsA("BasePart") and item.Name ~= "HumanoidRootPart" then
                capturePlayerPart(item)
                item.Color = Config.player_skin_color
                item.Material = mat
            end
            if item:IsA("MeshPart") or item:IsA("SpecialMesh") then
                capturePlayerMesh(item)
                pcall(function()
                    if item:IsA("MeshPart") then item.TextureID = "" else item.TextureId = "" end
                end)
            end
        end
    end
end

function updatePlayerSkin()
    if Config.custom_player_skin then applyPlayerSkin() else restorePlayerSkin() end
end

function getDynamicBoundingBox(character)
    local minX, minY, maxX, maxY = math.huge, math.huge, -math.huge, -math.huge
    local any = false
    for _, part in ipairs(character:GetChildren()) do
        if part:IsA("BasePart") then
            local sz = part.Size; local cf = part.CFrame
            local verts = {
                cf * Vector3.new(-sz.X/2, -sz.Y/2, -sz.Z/2),
                cf * Vector3.new( sz.X/2, -sz.Y/2, -sz.Z/2),
                cf * Vector3.new(-sz.X/2,  sz.Y/2, -sz.Z/2),
                cf * Vector3.new(-sz.X/2, -sz.Y/2,  sz.Z/2),
                cf * Vector3.new( sz.X/2,  sz.Y/2, -sz.Z/2),
                cf * Vector3.new( sz.X/2, -sz.Y/2,  sz.Z/2),
                cf * Vector3.new(-sz.X/2,  sz.Y/2,  sz.Z/2),
                cf * Vector3.new( sz.X/2,  sz.Y/2,  sz.Z/2)
            }
            for i = 1, 8 do
                local s, on = Camera:WorldToViewportPoint(verts[i])
                if on then any = true end
                if s.X < minX then minX = s.X end; if s.X > maxX then maxX = s.X end
                if s.Y < minY then minY = s.Y end; if s.Y > maxY then maxY = s.Y end
            end
        end
    end
    if not any then return nil end
    local w = maxX - minX; local h = maxY - minY
    if w < 5 or h < 5 then return nil end
    return minX, minY, w, h
end

                    else d.nameLabel.Visible = false end

                    if Config.health then
                        local hp = math.max(0, math.floor(hum.Health))
                        local maxHp = hum.MaxHealth > 0 and hum.MaxHealth or 100
                        local pct = math.clamp(hp / maxHp, 0, 1)
                        d.healthLabel.Text = hp .. " HP"
                        local col, trans
                        if Config.health_dynamic then
                            col = Config.health_low_color:Lerp(Config.health_high_color, pct)
                            trans = Config.health_low_transparency + (Config.health_high_transparency - Config.health_low_transparency) * pct
                        else
                            col = Config.health_color; trans = Config.health_transparency
                        end
                        d.healthLabel.TextColor3 = col
                        d.healthLabel.TextTransparency = trans
                        d.healthLabel.TextStrokeTransparency = math.clamp(trans + 0.3, 0, 1)
                        d.healthLabel.Position = UDim2.new(0, x + w + 5, 0, y + 1)
                        d.healthLabel.TextXAlignment = Enum.TextXAlignment.Left
                        d.healthLabel.Visible = true
                    else d.healthLabel.Visible = false end
                else
                    d.container.Visible = false; d.nameLabel.Visible = false; d.healthLabel.Visible = false
                end
            else
                if d then d.container.Visible = false; d.nameLabel.Visible = false; d.healthLabel.Visible = false end
            end
        end
    end
end

    alphaBg.Color = ColorSequence.new({ColorSequenceKeypoint.new(0,col), ColorSequenceKeypoint.new(1,col)})
    if activePreviewRect then
        activePreviewRect.BackgroundColor3 = col
        activePreviewRect.BackgroundTransparency = currentAlpha
    end
    if not skipHex then hexBox.Text = colorToHex(col) end
    if activeColorKey == "menu_color" then applyTheme()
    elseif activeColorKey == "lighting_color" then updateLighting()
    elseif activeColorKey == "fog_color" then updateAtmosphere()
    elseif string.sub(activeColorKey,1,11) == "skin_color_" then updateWeaponSkins()
    elseif activeColorKey == "player_skin_color" then updatePlayerSkin() end
end

function bindExternalColorPicker(colKey, alphaKey, preview, context)
    if activeColorKey == colKey then
        ColorPickerPanel.Visible = not ColorPickerPanel.Visible
        if not ColorPickerPanel.Visible then activeColorKey = nil; activeAlphaKey = nil; activePreviewRect = nil end
        return
    end
    activeColorKey = colKey; activeAlphaKey = alphaKey; activePreviewRect = preview
    CPIndicator.Text = context and context:upper() or "SELECTING COLOR"
    ColorPickerPanel.Visible = true
    local col = Config[colKey] or Color3.fromRGB(255,255,255)
    currentAlpha = Config[alphaKey] or 0
    currentH, currentS, currentV = Color3.toHSV(col)
    colorLayer.BackgroundColor3 = Color3.fromHSV(currentH,1,1)
    baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH,1,1))
    hueSlider.Position = UDim2.new(0,-2,currentH,0)
    pin.Position = UDim2.new(currentS,0,1-currentV,0)
    alphaSlider.Position = UDim2.new(1-currentAlpha,-1,0,-2)
    hexBox.Text = colorToHex(col)
    alphaBg.Color = ColorSequence.new({ColorSequenceKeypoint.new(0,col), ColorSequenceKeypoint.new(1,col)})
end

dragCanvas = false; dragHue = false; dragAlpha = false
canvas.InputBegan:Connect(function(input)
    if not activeColorKey then return end
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragCanvas = true
        local rx = math.clamp((input.Position.X - canvas.AbsolutePosition.X) / canvas.AbsoluteSize.X,0,1)
        local ry = math.clamp((input.Position.Y - canvas.AbsolutePosition.Y) / canvas.AbsoluteSize.Y,0,1)
        currentS = rx; currentV = 1-ry
        pin.Position = UDim2.new(rx,0,ry,0)
        updatePaletteColor(false)
    end
end)
hueStrip.InputBegan:Connect(function(input)
    if not activeColorKey then return end
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragHue = true
        local ry = math.clamp((input.Position.Y - hueStrip.AbsolutePosition.Y) / hueStrip.AbsoluteSize.Y,0,1)
        currentH = ry
        hueSlider.Position = UDim2.new(0,-2,ry,0)
        colorLayer.BackgroundColor3 = Color3.fromHSV(currentH,1,1)
        baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH,1,1))
        updatePaletteColor(false)
    end
end)
alphaStrip.InputBegan:Connect(function(input)
    if not activeColorKey then return end
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragAlpha = true
        local rx = math.clamp((input.Position.X - alphaStrip.AbsolutePosition.X) / alphaStrip.AbsoluteSize.X,0,1)
        currentAlpha = 1 - rx
        alphaSlider.Position = UDim2.new(rx,-1,0,-2)
        updatePaletteColor(false)
    end
end)
hexBox.FocusLost:Connect(function()
    if not activeColorKey then return end
    local c = hexToColor(hexBox.Text)
    if c then
        currentH, currentS, currentV = Color3.toHSV(c)
        colorLayer.BackgroundColor3 = Color3.fromHSV(currentH,1,1)
        baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH,1,1))
        hueSlider.Position = UDim2.new(0,-2,currentH,0)
        pin.Position = UDim2.new(currentS,0,1-currentV,0)
        updatePaletteColor(true)
        hexBox.Text = colorToHex(c)
    else
        hexBox.Text = colorToHex(Config[activeColorKey])
    end
end)
UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then dragCanvas = false; dragHue = false; dragAlpha = false end
end)
UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        if dragCanvas and ColorPickerPanel.Visible then
            local rx = math.clamp((input.Position.X - canvas.AbsolutePosition.X) / canvas.AbsoluteSize.X,0,1)
            local ry = math.clamp((input.Position.Y - canvas.AbsolutePosition.Y) / canvas.AbsoluteSize.Y,0,1)
            currentS = rx; currentV = 1-ry
            pin.Position = UDim2.new(rx,0,ry,0)
            updatePaletteColor(false)
        elseif dragHue and ColorPickerPanel.Visible then
            local ry = math.clamp((input.Position.Y - hueStrip.AbsolutePosition.Y) / hueStrip.AbsoluteSize.Y,0,1)
            currentH = ry
            hueSlider.Position = UDim2.new(0,-2,ry,0)
            colorLayer.BackgroundColor3 = Color3.fromHSV(currentH,1,1)
            baseGradient.Color = ColorSequence.new(Color3.fromHSV(currentH,1,1))
            updatePaletteColor(false)
        elseif dragAlpha and ColorPickerPanel.Visible then
            local rx = math.clamp((input.Position.X - alphaStrip.AbsolutePosition.X) / alphaStrip.AbsoluteSize.X,0,1)
            currentAlpha = 1 - rx
            alphaSlider.Position = UDim2.new(rx,-1,0,-2)
            updatePaletteColor(false)
        end
    end
end)

    local function spawn()
        local f = Instance.new("Frame")
        f.Size = UDim2.new(1,0,1,-42)
        f.Position = UDim2.new(0,0,0,42)
        f.BackgroundTransparency = 1
        f.Visible = false
        f.Parent = CombatPage
        return f, createPageList(f)
    end
    local p1, l1 = spawn(); local p2, l2 = spawn()
    p1.Visible = true
    local tabs = {}
    local function addSub(name, page, def)
        local btn = Instance.new("TextButton")
        btn.Size = UDim2.new(0,75,1,0)
        btn.BackgroundColor3 = def and Color3.fromRGB(28,28,28) or Color3.fromRGB(24,24,24)
        btn.BorderSizePixel = 0
        btn.Text = name
        btn.TextColor3 = def and Config.menu_color or Color3.fromRGB(140,140,150)
        btn.Font = def and Enum.Font.GothamMedium or Enum.Font.Gotham
        btn.TextSize = 11
        btn.Parent = header
        local ind = Instance.new("Frame")
        ind.Size = UDim2.new(0,4,1,0)
        ind.Position = UDim2.new(0,0,0,0)
        ind.BackgroundColor3 = Config.menu_color
        ind.BorderSizePixel = 0
        ind.Visible = def
        ind.Parent = btn
        local st = Instance.new("UIStroke")
        st.Thickness = 1
        st.Color = def and Config.menu_color or Color3.fromRGB(44,44,44)
        st.Parent = btn
        local info = {btn=btn, stroke=st, indicator=ind, page=page}
        tabs[name] = info
        table.insert(SubTabItemsTracking, info)
        btn.MouseButton1Click:Connect(function()
            for k, d in pairs(tabs) do
                local act = (k == name)
                d.page.Visible = act
                d.btn.TextColor3 = act and Config.menu_color or Color3.fromRGB(140,140,150)

                d.indicator.Visible = act
                d.stroke.Color = act and Config.menu_color or Color3.fromRGB(44,44,44)
            end
        end)
    end
    addSub("Aimbot", p1, true)
    addSub("Gun Mods", p2, false)
    return l1
end

    layout.Parent = header

    local function spawn()
        local f = Instance.new("Frame")
        f.Size = UDim2.new(1,0,1,-42)
        f.Position = UDim2.new(0,0,0,42)
        f.BackgroundTransparency = 1
        f.Visible = false
        f.Parent = MovementPage
        return f, createPageList(f)
    end
    local p1, l1 = spawn()
    p1.Visible = true
    local tabs = {}
    local function addSub(name, page, def)
        local btn = Instance.new("TextButton")
        btn.Size = UDim2.new(0,75,1,0)
        btn.BackgroundColor3 = def and Color3.fromRGB(28,28,28) or Color3.fromRGB(24,24,24)
        btn.BorderSizePixel = 0
        btn.Text = name
        btn.TextColor3 = def and Config.menu_color or Color3.fromRGB(140,140,150)
        btn.Font = def and Enum.Font.GothamMedium or Enum.Font.Gotham
        btn.TextSize = 11
        btn.Parent = header
        local ind = Instance.new("Frame")
        ind.Size = UDim2.new(0,4,1,0)
        ind.Position = UDim2.new(0,0,0,0)
        ind.BackgroundColor3 = Config.menu_color
        ind.BorderSizePixel = 0
        ind.Visible = def
        ind.Parent = btn
        local st = Instance.new("UIStroke")
        st.Thickness = 1
        st.Color = def and Config.menu_color or Color3.fromRGB(44,44,44)
        st.Parent = btn
        local info = {btn=btn, stroke=st, indicator=ind, page=page}
        tabs[name] = info
        table.insert(SubTabItemsTracking, info)
        btn.MouseButton1Click:Connect(function()
            for k, d in pairs(tabs) do
                local act = (k == name)
                d.page.Visible = act
                d.btn.TextColor3 = act and Config.menu_color or Color3.fromRGB(140,140,150)
                d.btn.Font = act and Enum.Font.GothamMedium or Enum.Font.Gotham

                d.indicator.Visible = act
                d.stroke.Color = act and Config.menu_color or Color3.fromRGB(44,44,44)
            end
        end)
    end
    addSub("Anti Aim", p1, true)
    return l1
end

-- ==================== MENU TOGGLE ====================
transitionActive = false
function toggleMenu()
    if transitionActive then return end
    transitionActive = true
    menuVisible = not menuVisible
    if menuVisible then
        MainFrame.Visible = true
        TweenService:Create(MainFrame, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        TweenService:Create(Sidebar, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        TweenService:Create(Logo, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {TextTransparency = 0}):Play()
        TweenService:Create(MainStroke, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0}):Play()
        TweenService:Create(SepLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        TweenService:Create(AccentLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
        task.wait(0.15)
    else
        TweenService:Create(MainFrame, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        TweenService:Create(Sidebar, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        TweenService:Create(Logo, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {TextTransparency = 1}):Play()
        TweenService:Create(MainStroke, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {Transparency = 1}):Play()
        TweenService:Create(SepLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        TweenService:Create(AccentLine, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1}):Play()
        task.wait(0.15)
        if not menuVisible then
            MainFrame.Visible = false
            ColorPickerPanel.Visible = false
            activeColorKey = nil; activeAlphaKey = nil; activePreviewRect = nil
        end
    end
    transitionActive = false
end

table.insert(connections, RunService.RenderStepped:Connect(function(dt)
    processVisuals()
    processAimbot()
    processFakelagEngineRenderCycle(dt)
    updateHUDOverlay(dt)
    updateArrayList()

    shineOffsetProgress = shineOffsetProgress + dt * 0.8
    if shineOffsetProgress > 1.2 then shineOffsetProgress = -1.2 end
    local off = Vector2.new(shineOffsetProgress, 0)
    for i = #globalShineGradients, 1, -1 do
        local g = globalShineGradients[i]
        if not g or not g.Parent then
            table.remove(globalShineGradients, i)
        else
            if g.Parent.Visible then g.Offset = off end
        end
    end
end))

-- Input handling
table.insert(connections, UserInputService.InputBegan:Connect(function(input, processed)
    if UserInputService:GetFocusedTextBox() then return end

    if UIRefs.listeningForBind and input.UserInputType == Enum.UserInputType.Keyboard then
        Config.toggleKey = input.KeyCode
        UIRefs.BindButton.Text = input.KeyCode.Name
        UIRefs.BindButton.TextColor3 = Config.menu_color
        UIRefs.listeningForBind = false
        return
    end

    if listeningForAimbotBind then
        local bound = nil
        if input.UserInputType == Enum.UserInputType.Keyboard and input.KeyCode ~= Enum.KeyCode.Unknown then
            bound = input.KeyCode
        elseif input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.MouseButton2 or input.UserInputType == Enum.UserInputType.MouseButton3 then
            bound = input.UserInputType
        end
        if bound then
            Config.aimbot_key = bound
            UIRefs.AimbotBindBtn.Text = getBindName(bound)
            UIRefs.AimbotBindBtn.TextColor3 = Config.menu_color
            listeningForAimbotBind = false
            aimbotKeyPressed = false
            return
        end
    end

    if listeningForFakelagBind then
        local bound = nil
        if input.UserInputType == Enum.UserInputType.Keyboard and input.KeyCode ~= Enum.KeyCode.Unknown then
            bound = input.KeyCode
        elseif input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.MouseButton2 or input.UserInputType == Enum.UserInputType.MouseButton3 then
            bound = input.UserInputType
        end
        if bound then
            Config.fakelag_key = bound
            UIRefs.FakelagBindBtn.Text = getBindName(bound)
            UIRefs.FakelagBindBtn.TextColor3 = Config.menu_color
            listeningForFakelagBind = false
            fakelagKeyPressed = false
            return
        end
    end

    if processed then return end

    if input.KeyCode == Config.toggleKey then toggleMenu() end

    if matchesBind(input, Config.aimbot_key) then
        if Config.aimbot_mode == "Hold" then aimbotKeyPressed = true
        elseif Config.aimbot_mode == "Toggle" then aimbotKeyPressed = not aimbotKeyPressed end
    end

    if matchesBind(input, Config.fakelag_key) then
        if Config.fakelag_mode == "Hold" then fakelagKeyPressed = true
        elseif Config.fakelag_mode == "Toggle" then fakelagKeyPressed = not fakelagKeyPressed end
    end
end))

table.insert(connections, UserInputService.InputEnded:Connect(function(input, processed)
    if UserInputService:GetFocusedTextBox() then return end
    if matchesBind(input, Config.aimbot_key) and Config.aimbot_mode == "Hold" then
        aimbotKeyPressed = false
    end
    if matchesBind(input, Config.fakelag_key) and Config.fakelag_mode == "Hold" then
        fakelagKeyPressed = false
    end
end))
